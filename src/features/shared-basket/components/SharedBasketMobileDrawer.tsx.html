  SHARED_BASKET_DRAWER_COPY,
} from '../constants/shareBasket.constants';
 
// ─── Local composite type (relaxed totalPrice — no CentPrecisionMoney required) ─
interface CompositeCartItem {
  id: string;
  title?: string | Record<string, string>;
  name?: string | Record<string, string>;
  quantity: number;
  currencyCode?: string;
  startDate?: string;
  endDate?: string;
  totalPrice?: { centAmount: number; currencyCode?: string };
  addOns?: string[];
  delegateRate: string;
}
 
// ─── Constants ────────────────────────────────────────────────────────────────
 
const SUB_TEXT_COLOUR = 'rgba(0,43,102,0.7)' as const;
 
// ─── Props ────────────────────────────────────────────────────────────────────
 
interface SharedBasketMobileDrawerProps {
  open: boolean;
  onClose: () => void;
}

const SharedBasketMobileDrawer = ({ open, onClose }: SharedBasketMobileDrawerProps) => {
  const navigate = useNavigate();
  const params = useParams<{ storeKey?: string; eventKey?: string }>();
  const [context] = useAppContext();
  const activeKeys = getActiveKeys();
 
  const storeKey = params.storeKey ?? activeKeys.storeKey;
  const eventKey = params.eventKey ?? activeKeys.eventKey;
 
  const currency =
    context?.currency || sessionStorage.getItem(STORAGE_KEYS.CURRENCY) || CART_DEFAULTS.CURRENCY;
  const locale =
    context?.locale || sessionStorage.getItem(STORAGE_KEYS.LOCALE) || CART_DEFAULTS.LOCALE;
 
  const { items, cart } = useMiniBasketData();
  const basketActions = useMiniBasketActions(items, cart);
 
  const scrollRef = useRef<HTMLDivElement>(null);
 
  // Prevent body scroll while drawer is open
  useEffect(() => {
    if (open) {
      document.body.style.overflow = 'hidden';
    } else {
      document.body.style.overflow = '';
    }
    return () => {
      document.body.style.overflow = '';
    };
  }, [open]);
 
  // Close on Escape key
  useEffect(() => {
    if (!open) return;
    const handleKeyDown = (e: KeyboardEvent) => {
      if (e.key === KEYBOARD_KEYS.ESCAPE) onClose();
    };
    document.addEventListener(KEYDOWN_EVENT, handleKeyDown);
    return () => document.removeEventListener(KEYDOWN_EVENT, handleKeyDown);
  }, [open, onClose]);
 
  const handleViewFullOrder = () => {
    onClose();
    navigate(`/${storeKey}/cart`);
  };
 
  const handleCheckout = () => {
    onClose();
    const pingToken = sessionStorage.getItem(STORAGE_KEYS.PING_ACCESS_TOKEN);
    const isLoggedIn = context?.loggedIn || !!pingToken;
    if (!isLoggedIn) {
      const redirectPath = `/${storeKey}/checkout`;
      localStorage.setItem(STORAGE_KEYS.POST_LOGIN_REDIRECT, redirectPath);
      navigate(`/${storeKey}/${eventKey}/account?returnUrl=${encodeURIComponent(redirectPath)}`);
    } else {
      navigate(`/${storeKey}/checkout`);
    }
  };
 
  // Derive grouped display items (mirrors MiniBasketDropdown logic)
  const { groups: basketGroups, updatingItems } = basketActions;
 
  const groups = useMemo(() => {
    const derivedGroups: Array<{
      compositeItem: CompositeCartItem;
      groupItems: MiniBasketDisplayItem[];
      groupKey: string;
      isUpdating: boolean;
    }> = [];
 
    Object.entries(basketGroups).forEach(([key, groupItems]) => {
      const effectiveMainItems = groupItems.filter((i) => isPassProduct(i));
      const effectiveMainItem = effectiveMainItems[0] || groupItems[0];
      const totalQuantity =
        effectiveMainItems.reduce((sum, i) => sum + i.quantity, 0) || effectiveMainItem.quantity;
      const mainIds = new Set(effectiveMainItems.map((i) => i.id));
      const addOnItems = groupItems.filter((i) => !mainIds.has(i.id));
 
      const { custom: _custom, ...restItems } = effectiveMainItem;
      const compositeItem = {
        ...restItems,
        quantity: totalQuantity,
        startDate:
          (effectiveMainItem.variant?.attributes?.find(
            (a: { name: string; value: unknown }) => a.name === CUSTOM_FIELD_START_DATE,
          )?.value as string | undefined) ||
          (effectiveMainItem.custom?.fields?.[CUSTOM_FIELD_START_DATE] as string | undefined) ||
          effectiveMainItem.startDate,
        endDate:
          (effectiveMainItem.variant?.attributes?.find(
            (a: { name: string; value: unknown }) => a.name === CUSTOM_FIELD_END_DATE,
          )?.value as string | undefined) ||
          (effectiveMainItem.custom?.fields?.[CUSTOM_FIELD_END_DATE] as string | undefined) ||
          effectiveMainItem.endDate,
        totalPrice: {
          centAmount: groupItems.reduce((sum, i) => sum + (i.totalPrice?.centAmount || 0), 0),
          currencyCode: effectiveMainItem.totalPrice?.currencyCode || CART_DEFAULTS.CURRENCY,
        },
        addOns: [
          ...new Set(
            addOnItems.map((i) => {
              if (!i.name) return 'Add on';
              if (typeof i.name === 'string') return i.name;
              const nameObj = i.name as Record<string, string>;
              if (nameObj[CART_DEFAULTS.LOCALE]) return nameObj[CART_DEFAULTS.LOCALE];
              const currentLoc = sessionStorage.getItem(STORAGE_KEYS.LOCALE);
              if (currentLoc && nameObj[currentLoc]) return nameObj[currentLoc];
              const values = Object.values(nameObj);
              return values.length > 0 ? values[0] : 'Add on';
            }),
          ),
        ].sort(),
        delegateRate:
          (groupItems[0].custom?.fields?.[CUSTOM_FIELD_DELEGATE_RATE] as string | undefined) || '',
      };
 
      const firstItem = groupItems[0];
      const assocKey = firstItem.custom?.fields?.[CUSTOM_FIELD_ASSOCIATION_KEY] as string | undefined;
      const delegateRate = compositeItem.delegateRate;
      const gKey =
        assocKey && delegateRate
          ? `${assocKey}||${delegateRate}`
          : (assocKey ||
            groupItems
              .map((i) => i.id)
              .sort()
              .join('_'));
 
      derivedGroups.push({
        compositeItem,
        groupItems,
        groupKey: key,
        isUpdating: updatingItems.has(gKey),
      });
    });
 
    return derivedGroups;
  }, [basketGroups, updatingItems]);
 
  const totalGroupedCount = groups.reduce((s, g) => s + g.compositeItem.quantity, 0);
  const itemCount = totalGroupedCount + basketActions.looseItems.length;
 
  if (!open) return null;
 
  return (
    /* Mobile-only: md:hidden keeps this off desktop/tablet */
    <div className="md:hidden fixed inset-0 z-[60] flex flex-col justify-end">
      {/* Backdrop */}
      <div className="absolute inset-0 bg-black/40" onClick={onClose} aria-hidden="true" />
 
      {/* Drawer panel */}
      <div className="relative z-10 w-full bg-white rounded-t-[12px] flex flex-col max-h-[85vh] shadow-[0px_-4px_20px_rgba(0,0,0,0.12)]">
        {/* ── Header ──────────────────────────────────────────────────────── */}
        <div className="flex-shrink-0 flex items-center justify-between px-3 py-3 border-b border-[#EAECF0] rounded-tl-[12px] rounded-tr-[12px] bg-white">

            type="button"

          >
            <img src={CloseIcon} alt="" aria-hidden="true" className="w-4 h-4" />
          </button>
        </div>
 
        {/* ── Scrollable items list ────────────────────────────────────────── */}
        <div
          ref={scrollRef}
          className="flex-1 overflow-y-auto bg-[#F2F4F7] px-1.5 pt-1.5 pb-3 min-h-0"
        >
          {groups.length === 0 && basketActions.looseItems.length === 0 ? (
            <div className="flex items-center justify-center h-24">
              <p className="text-sm text-[rgba(0,43,102,0.7)] font-sans">{SHARED_BASKET_DRAWER_COPY.EMPTY}</p>
            </div>
          ) : (

                      Object.values(compositeItem.title || {})[0] ||
                      '';

                const delegateRate = compositeItem.delegateRate;

                  <div
                    key={`group-${groupKey}`}
                    className="bg-white rounded-md p-3 flex flex-col gap-3"
                  >
                    {/* Pass name + meta */}
                    <div className="flex flex-col gap-0">
                      {/* Title */}
                      <p className="text-[#002B66] text-sm font-semibold font-sans leading-6 line-clamp-2 m-0">
                        {title}
                      </p>
                      {/* Meta row */}
                      <div className="flex flex-wrap items-center gap-x-2 gap-y-0.5 mt-0.5">

                          <DateRange
                            startDate={compositeItem.startDate}
                            endDate={compositeItem.endDate}

                            format="MiniBasket"
                            className="flex items-center gap-1"
                            icon={

                            <img
                              src={AddOnIcon}
                              alt=""
                              className="w-3 h-3 shrink-0"
                              aria-hidden="true"
                            />
                            <span
                              className="text-xs font-normal font-sans line-clamp-1"
                              style={{ color: SUB_TEXT_COLOUR }}
                            >
                              {addOnCount}{' '}
                              {addOnCount === 1
                                ? SHARED_BASKET_DRAWER_COPY.ADD_ON_SINGULAR
                                : SHARED_BASKET_DRAWER_COPY.ADD_ON_PLURAL}

                const totalCents = item.totalPrice?.centAmount ?? 0;

                  (item.custom?.fields?.[CUSTOM_FIELD_ASSOCIATION_KEY] as string | undefined) ||
                  item.id;
                const isUpdating = updatingItems.has(gKey);
 
                return (
                  <div key={item.id} className="bg-white rounded-md p-3 flex flex-col gap-3">
                    <p className="text-[#002B66] text-sm font-semibold font-sans leading-6 line-clamp-2 m-0">
                      {title}
                    </p>
                    <div className="flex items-center justify-between">
                      <QuantityControl

                        disabled={isUpdating}
                        variant={QUANTITY_VARIANTS.MINI_BASKET}
                        onIncrease={() => basketActions.handleIncrement(item.id)}
                        onDecrease={() => basketActions.handleDecrement(item.id)}

                      <span className="text-[#002B66] text-base font-semibold font-sans leading-6 text-right">
                        {price}
                      </span>
                    </div>
                  </div>
                );
              })}
            </div>
          )}
        </div>
 
        {/* ── Footer ──────────────────────────────────────────────────────── */}
        <div className="flex-shrink-0 bg-white px-3 pt-3 pb-[max(12px,env(safe-area-inset-bottom))] flex flex-col gap-3 shadow-[0px_-1px_20px_0px_rgba(0,0,0,0.08)] rounded-tl-md rounded-tr-md">
          {/* Total row */}
          <div className="flex flex-col gap-0.5">
            <div className="flex items-baseline gap-3 w-full">
              <span className="text-[#002B66] text-sm font-medium font-sans leading-6">
                {SHARED_BASKET_DRAWER_COPY.TOTAL}
              </span>
              <div className="flex-1 flex justify-end">
                <span className="text-[#002B66] text-base font-semibold font-sans leading-6 text-right">
                  {formatCurrency(basketActions.subtotal, currency, locale)}
                </span>
              </div>
            </div>
            <p className="text-right text-[rgba(0,43,102,0.7)] text-xs font-normal font-sans leading-4 m-0">
              {SHARED_BASKET_DRAWER_COPY.TAX}
            </p>
          </div>

          <div className="flex items-center gap-2 w-full">
            {/* View Full Order — outline primary */}
            <button

              <span className="text-[#0A51C2] text-xs font-bold font-sans leading-4 text-center">
                {SHARED_BASKET_DRAWER_COPY.VIEW_FULL_ORDER}
              </span>
            </button>
 
            {/* Checkout — solid primary */}
            <button
              type="button"
              onClick={handleCheckout}
              className="flex-1 h-10 px-4 bg-[#0A51C2] hover:bg-[#0947AB] active:bg-[#073A8C] rounded-md flex items-center justify-center gap-1.5 overflow-hidden transition-colors"
            >
              <span className="text-white text-xs font-bold font-sans leading-4 text-center">
                {SHARED_BASKET_DRAWER_COPY.CHECKOUT}

            </button>
          </div>
        </div>
      </div>
    </div>
  );
};
