/**
 * HotelsNearby — "Check hotels nearby" call-to-action card.
 *
 * Renders on the Order Confirmation and Invoice Confirmation pages.
 *
 * Responsive behaviour (driven by parent layouts):
 *  - Spans self-stretch to fit the parent container width.
 *  - Height is determined dynamically based on content (image height 80px + content padding).
 *
 * All dimensions, colours, typography and spacing values match the Figma spec
 * exactly. No inline styles — Tailwind utility classes only, using custom CSS variable fallbacks.
 */
import React from 'react';
 
import hotelImage from '../../../assets/icons/order-confirmation/hotel_image.svg';
import Button from '../../../shared/components/Button';
import { ORDER_TEXTS, HOTEL_MAP_CONSTANTS } from '../constants/order.constants';
 
interface HotelsNearbyProps {
    /** URL to open when the "View hotels" button is clicked. Defaults to '#'. */
    bookingUrl?: string;
}
 
const ensureAbsoluteUrl = (url: string) => {
    if (!url || url === HOTEL_MAP_CONSTANTS.DEFAULT_HASH) return url;
    // Check if URL already has a protocol or starts with //
    if (/^https?:\/\//i.test(url) || url.startsWith('//')) {
        return url;
    }
    return `https://${url}`;
};
 
const HotelsNearby: React.FC<HotelsNearbyProps> = ({ bookingUrl = HOTEL_MAP_CONSTANTS.DEFAULT_HASH }) => {
    const absoluteUrl = ensureAbsoluteUrl(bookingUrl);
 
    return (
        <div className="self-stretch w-full max-w-[335px] tablet:w-[332px] min-[1025px]:w-[267px] mx-auto tablet:ml-0 tablet:mr-auto rounded-md shadow-[0px_1px_2px_0px_rgba(12,17,29,0.15)] flex flex-col justify-start items-start overflow-hidden bg-[var(--colours-surface-neutral-light-white,#FFFFFF)]">
            <img className="w-full h-20 object-cover rounded-t-md block" src={hotelImage} alt="" />
            <div className="self-stretch p-5 bg-[var(--colours-surface-neutral-light-white,#FFFFFF)] flex flex-col justify-start items-start gap-5">
                <div className="self-stretch flex flex-col justify-start items-start gap-0.5">
                    <h3 className="self-stretch justify-end text-[var(--colours-text-default-primary-heading,#002B66)] text-base font-semibold font-['Noto_Sans'] leading-6 align-bottom">
                        {ORDER_TEXTS.HOTEL_MAP_TITLE}
                    </h3>
                    <p className="self-stretch justify-end text-[var(--colours-text-default-primary-body-base,#002B66B2)] text-sm font-normal font-['Noto_Sans'] leading-6 align-bottom">
                        {ORDER_TEXTS.HOTEL_MAP_DESC}
                    </p>
                </div>
                <Button
                    type="button"
                    onClick={() => {
                        if (absoluteUrl !== HOTEL_MAP_CONSTANTS.DEFAULT_HASH) {
                            window.open(absoluteUrl, HOTEL_MAP_CONSTANTS.TARGET_BLANK, HOTEL_MAP_CONSTANTS.WINDOW_FEATURES);
                        }
                    }}
                    className="!h-9 px-4 py-2 !bg-[var(--component-button-primary-solid-default-surface,#0A51C2)] rounded-md inline-flex justify-center items-center gap-1.5 overflow-hidden"
                    aria-label={ORDER_TEXTS.HOTEL_MAP_ARIA}
                >
                    <span className="text-center justify-start text-[var(--component-button-primary-solid-default-text,#FFFFFF)] text-xs font-bold font-['Noto_Sans'] leading-4">
                        {ORDER_TEXTS.HOTEL_MAP_BUTTON}
                    </span>
                </Button>
            </div>
        </div>
    );
};
 
export default HotelsNearby;
