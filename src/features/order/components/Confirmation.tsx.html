                                            className="justify-end font-noto font-medium text-[12px] leading-[18px] tracking-[0px] text-[var(--colours-text-default-primary-heading,#002B66)] w-auto whitespace-nowrap h-[18px] align-bottom"
                                        >
                                            {eventName}
                                        </div>
                                        <div className="flex justify-start items-center gap-[4px]">
                                            <img
                                                src={eventLogo}
                                                alt=""
                                                className="w-[51px] h-[20px] opacity-100 rotate-0"
                                            />
                                        </div>
                                    </div>
 
                                    {/* Passes List / Details */}
                                    <div className="self-stretch flex flex-col justify-start items-start gap-6">
                                        {displayProducts.map((product, index) => (
                                            <div key={product.id || index} className="self-stretch flex flex-col justify-start items-start gap-2">
                                                <div className="self-stretch flex flex-col justify-start items-start gap-1">
                                                    {/* Date and Standard (Metadata) Layout Row (Row 1 - Top) */}
                                                    <div className="self-stretch flex flex-row flex-wrap tablet:flex-nowrap justify-start items-center gap-[8px] h-auto tablet:h-[18px]">
                                                        {/* Quantity */}
                                                        <div className="flex justify-start items-center gap-[4px] h-[18px]">
                                                            <div className="w-[12px] h-[12px] relative overflow-hidden flex-shrink-0">
                                                                <img src={confirmationNumberIcon} alt="" className="w-[12px] h-[18px] absolute left-0 top-[-3px] object-contain" />
                                                            </div>
                                                            <div
                                                                className="font-noto font-normal text-[12px] leading-[18px] tracking-[0px] text-[var(--colours-text-default-primary-heading,#002B66)] whitespace-nowrap"
                                                            >
                                                                {product.quantity || 1}
                                                            </div>
                                                        </div>
 
                                                        {/* Dates */}
                                                        {(product.startDate || product.date) && (
                                                            <div className="flex justify-start items-center gap-[4px] h-[18px]">
                                                                <div className="w-[12px] h-[12px] relative overflow-hidden flex-shrink-0">
                                                                    <img src={calendarTodayIcon} alt="" className="w-[12px] h-[18px] absolute left-0 top-[-3px] object-contain" />
                                                                </div>
                                                                <div
                                                                    className="font-noto font-normal text-[12px] leading-[18px] tracking-[0px] text-[var(--colours-text-default-primary-heading,#002B66)] whitespace-nowrap"
                                                                >
                                                                    {product.startDate ? (
                                                                        <DateRange
                                                                            startDate={product.startDate}
                                                                            endDate={product.endDate}
                                                                            className="font-noto font-normal text-[12px] leading-[18px] tracking-[0px] text-[var(--colours-text-default-primary-heading,#002B66)]"
                                                                        />
                                                                    ) : (
                                                                        product.date
                                                                    )}
                                                                </div>
                                                            </div>
                                                        )}
 
                                                        {/* Delegate Rate / Ticket Type */}
                                                        <div className="flex justify-start items-center gap-[4px] h-[18px]">
                                                            <div className="w-[12px] h-[12px] relative overflow-hidden flex-shrink-0">
                                                                <img src={userAttributesIcon} alt="" className="w-[12px] h-[18px] absolute left-0 top-[-3px] object-contain" />
                                                            </div>
                                                            <div
                                                                className="font-noto font-normal text-[12px] leading-[18px] tracking-[0px] text-[var(--colours-text-default-primary-heading,#002B66)] whitespace-nowrap"
                                                            >
                                                                {product.delegateRate || ORDER_TEXTS.DEFAULT_DELEGATE_RATE}
                                                            </div>
                                                        </div>
                                                    </div>
                                                    {/* Pass Name & Price Row (Row 2 - Bottom) */}
                                                    <div className="self-stretch flex flex-col tablet:flex-row justify-between items-start gap-2 tablet:gap-[8px] tablet:h-[26px]">
                                                        {/* Pass Name */}
                                                        <div
                                                            className="w-full tablet:flex-1 tablet:min-w-0 font-noto font-semibold text-[16px] leading-[24px] tablet:text-[18px] tablet:leading-[26px] tracking-[0px] text-[var(--colours-text-default-primary-heading,#002B66)] truncate"
                                                        >
                                                            {product.name}
                                                        </div>
                                                        {/* Price */}
                                                        <div
                                                            className="w-auto h-[24px] font-noto font-semibold text-[16px] leading-[24px] tracking-[0px] text-right whitespace-nowrap text-[var(--colours-text-default-primary-heading,#002B66)]"
                                                        >
                                                            {formatPrice(product.totalPrice ?? ((product.price ?? 0) * (product.quantity ?? 1)), currencyCode)}
                                                        </div>
                                                    </div>
                                                </div>
 
                                                {/* Add-ons */}
                                                {product.addons && product.addons.length > 0 && (
                                                    <div className="self-stretch flex flex-col justify-start items-start gap-1">
                                                        {product.addons.map((addon) => (
                                                            <div
                                                                key={addon.id}
                                                                className="self-stretch flex flex-row justify-between items-start gap-[8px] h-[24px]"
                                                            >
                                                                <div
                                                                    className="flex-1 text-left font-noto font-normal text-[14px] leading-[24px] text-[var(--colours-text-default-primary-heading,#002B66)] truncate"
                                                                >
                                                                    {addon.name}
                                                                </div>
                                                                <div
                                                                    className="w-auto h-[24px] text-right whitespace-nowrap font-noto font-normal text-[14px] leading-[24px] text-[var(--colours-text-default-primary-heading,#002B66)]"
                                                                >
                                                                    {formatPrice(addon.totalPrice ?? ((addon.price ?? 0) * (addon.quantity ?? 1)), currencyCode)}
                                                                </div>
                                                            </div>
                                                        ))}
                                                    </div>
                                                )}
                                            </div>
                                        ))}
                                    </div>
 
                                    {/* Divider above Summary Block */}
                                    <div className="self-stretch h-0 border-t border-[var(--colours-border-neutral-lighter,#F0F2F5)]" />
 
                                    {/* Summary Block */}
                                    <div className="self-stretch flex flex-col justify-start items-start gap-0.5">
                                        {/* Total Discount */}
                                        {discountCentAmount > 0 && (
                                            <div className="self-stretch inline-flex justify-start items-center gap-2">
                                                <div
                                                    className="flex-1 justify-start text-sm font-normal font-['Noto_Sans'] leading-6 line-clamp-1 text-[var(--colours-text-default-primary-heading,#002B66)]"
                                                >
                                                    {ORDER_SUMMARY.TOTAL_DISCOUNT_LABEL}
                                                </div>
                                                <div className="flex justify-start items-center gap-1">
                                                    <div
                                                        className="text-right justify-start text-sm font-normal font-['Noto_Sans'] leading-6 text-[var(--colours-text-default-primary-heading,#002B66)]"
                                                    >
                                                        -{formatPrice(discountCentAmount, currencyCode)}
                                                    </div>
                                                </div>
                                            </div>
                                        )}
 
                                        {/* Sub Total */}
                                        <div className="self-stretch inline-flex justify-start items-center gap-2">
                                            <div
                                                className="flex-1 justify-start text-sm font-normal font-['Noto_Sans'] leading-6 line-clamp-1 text-[var(--colours-text-default-primary-heading,#002B66)]"
                                            >
                                                {ORDER_SUMMARY.SUB_TOTAL_LABEL}
                                            </div>
                                            <div
                                                className="text-right justify-start text-sm font-normal font-['Noto_Sans'] leading-6 text-[var(--colours-text-default-primary-heading,#002B66)]"
                                            >
                                                {formatPrice(subtotalCentAmount, currencyCode)}
                                            </div>
                                        </div>
 
                                        {/* Tax rows — one per authority */}
                                        {aggregatedTaxLines.map((line) => {
                                            const ratePercent = (line.taxRate * 100).toFixed(2).replace(/\.?0+$/, '');
                                            const label = line.hideRate
                                                ? `Total ${line.authorityType}:`
                                                : `Total ${line.authorityType} (${ratePercent}%):`;
                                            return (
                                                <div key={line.authorityType} className="self-stretch inline-flex justify-start items-center gap-2">
                                                    <div
                                                        className="flex-1 justify-start text-sm font-normal font-['Noto_Sans'] leading-6 line-clamp-1 text-[var(--colours-text-default-primary-heading,#002B66)]"
                                                    >
                                                        {label}
                                                    </div>
                                                    <div
                                                        className="text-right justify-start text-sm font-normal font-['Noto_Sans'] leading-6 text-[var(--colours-text-default-primary-heading,#002B66)]"
                                                    >
                                                        {formatPrice(line.taxAmount * DISCOUNT_BREAKDOWN_CONSTANTS.CENTS_CONVERSION_FACTOR, currencyCode)}
                                                    </div>
                                                </div>
                                            );
                                        })}
                                    </div>
 
                                    {/* Total Amount Row */}
                                    <div className="self-stretch inline-flex justify-start items-start gap-2">
                                        <div
                                            className="flex-1 justify-start text-[20px] tablet:text-[22px] font-semibold font-['Noto_Sans'] leading-[28px] tablet:leading-[32px] line-clamp-2 text-[var(--colours-text-default-primary-heading,#002B66)]"
                                        >
                                            {ORDER_SUMMARY.TOTAL}
                                        </div>
                                        <div
                                            className="text-right justify-start text-[20px] tablet:text-[22px] font-semibold font-['Noto_Sans'] leading-[28px] tablet:leading-[32px] text-[var(--colours-text-default-primary-heading,#002B66)]"
                                        >
                                            {formatPrice(displayedTotalCentAmount, currencyCode)}
                                        </div>
                                    </div>
                                </div>
                            </div>
                        ) : (
                            <div className="bg-white border border-gray-300 rounded-[8px] p-6 text-gray-500 text-center">
                                {ORDER_TEXTS.NO_BOOKING_DETAILS}
                            </div>
                        )}
 
                        {/* Tablet / mobile hotels slot (visible < 1025px) */}
                        {hotelsNearby && (
                            <div className="min-[1025px]:hidden">
                                {hotelsNearby}
                            </div>
                        )}
                    </div>
                    {/* End left column */}
 
                    {/* ── Sidebar Column ── */}
                    {sidebar && (
                        <div className="h-full">
                            {sidebar}
                        </div>
                    )}
                </div>
                {/* End grid */}
            </div>
            {/* End max-w wrapper */}
        </div>
    );
};
