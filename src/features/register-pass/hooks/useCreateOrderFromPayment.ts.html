import { useState, useEffect, useRef } from 'react';
 
import { PAYMENT_INTENT, PAYMENT_CREATE } from '../../../shared/constants/api';
import * as api from '../../../shared/services/apiClient';
import { PAYMENT_METHOD, ORDER_FIELD_KEYS, ORDER_CT_ACTIONS, ORDER_CT_TYPES, ORDER_STORAGE_KEYS } from '../../order/constants/order.constants';
import { ensureOrderExists, getOrderById, updateOrderCustomFields } from '../../order/services/order.service';
import type { OrderWithPayment } from '../../order/types/order.types';
import { TAX_STORAGE_KEYS } from '../../tax/constants/tax.constants';
 
export const useCreateOrderFromPayment = (paymentIntentId: string | null, invoiceOrderId: string | null = null) => {
    const [loading, setLoading] = useState(false);
    const [error, setError] = useState<string | null>(null);
    const [orderId, setOrderId] = useState<string | null>(null);
    const processedRef = useRef(false);
 
    useEffect(() => {
        if ((!paymentIntentId && !invoiceOrderId) || processedRef.current) return;
 
        const createOrder = async () => {
            processedRef.current = true;
            setLoading(true);
            try {
                let currentOrder: OrderWithPayment | null = null;
                let cartId: string | null = null;
                let cartVersion: string | number | null = null;
 
                if (paymentIntentId) {
                    const res = await api.get(PAYMENT_INTENT(paymentIntentId));
                    if (!res.ok) throw new Error('Failed to retrieve payment intent');
                    const data = (await res.json()) as { metadata?: { cartId?: string; cartVersion?: string | number } };
                    cartId = data.metadata?.cartId ?? null;
                    cartVersion = data.metadata?.cartVersion ?? null;
 
                    if (cartId) {
                        currentOrder = (await ensureOrderExists(cartId, cartVersion, [], true)) as OrderWithPayment | null;
 
                        if (currentOrder && paymentIntentId) {
                            const payments = currentOrder.paymentInfo?.payments;
                            const hasPayment = Array.isArray(payments) && payments.length > 0;
 
                            if (!hasPayment) {
                                try {
                                    const amountCent =
                                        currentOrder.taxedPrice?.totalGross?.centAmount ||
                                        currentOrder.totalPrice?.centAmount ||
                                        0;
                                    const currencyCode =
                                        currentOrder.taxedPrice?.totalGross?.currencyCode ||
                                        currentOrder.totalPrice?.currencyCode ||
                                        ORDER_FIELD_KEYS.CURRENCY_AUTO;
 
                                    const paymentDraft = {
                                        amountPlanned: { currencyCode, centAmount: amountCent },
                                        paymentMethodInfo: { paymentInterface: PAYMENT_METHOD.STRIPE, method: PAYMENT_METHOD.CARD },
                                        interfaceId: paymentIntentId,
                                    };
 
                                    const paymentRes = await api.post(PAYMENT_CREATE, paymentDraft);
                                    if (paymentRes.ok) {
                                        const payment = await paymentRes.json();
                                        currentOrder = (await updateOrderCustomFields(currentOrder as unknown as Record<string, unknown>, [
                                            {
                                                action: ORDER_CT_ACTIONS.ADD_PAYMENT,
                                                payment: { typeId: ORDER_CT_TYPES.PAYMENT, id: payment.id }
                                            }
                                        ])) as unknown as OrderWithPayment;
                                    }
                                } catch (payErr) {
                                    console.error('Failed to process consolidated card payment:', payErr);
                                }
                            }
                        } else if (currentOrder) {
                            currentOrder = (await updateOrderCustomFields(currentOrder as unknown as Record<string, unknown>)) as unknown as OrderWithPayment;
                        }
                    }
                } else if (invoiceOrderId) {
                    currentOrder = (await getOrderById(invoiceOrderId)) as OrderWithPayment | null;
                    if (currentOrder) {
                        const payments = currentOrder.paymentInfo?.payments;
                        const hasPayment = Array.isArray(payments) && payments.length > 0;
 
                        if (!hasPayment) {
                            try {
                                const amountCent = currentOrder.taxedPrice?.totalGross?.centAmount || currentOrder.totalPrice?.centAmount || 0;
                                const currencyCode = currentOrder.taxedPrice?.totalGross?.currencyCode || currentOrder.totalPrice?.currencyCode || ORDER_FIELD_KEYS.CURRENCY_AUTO;
 
                                const paymentDraft = {
                                    amountPlanned: { currencyCode, centAmount: amountCent },
                                    paymentMethodInfo: { paymentInterface: '', method: PAYMENT_METHOD.INVOICE },
                                };
 
                                const paymentRes = await api.post(PAYMENT_CREATE, paymentDraft);
                                if (paymentRes.ok) {
                                    const payment = await paymentRes.json();
                                    currentOrder = (await updateOrderCustomFields(currentOrder as unknown as Record<string, unknown>, [
                                        {
                                            action: ORDER_CT_ACTIONS.ADD_PAYMENT,
                                            payment: { typeId: ORDER_CT_TYPES.PAYMENT, id: payment.id }
                                        }
                                    ])) as unknown as OrderWithPayment;
                                }
                            } catch (payErr) {
                                console.error('Failed to process invoice payment linking:', payErr);
                            }
                        }
                    }
                }
 
                if (currentOrder?.id) {
                    setOrderId(currentOrder.id);
                    sessionStorage.setItem(ORDER_STORAGE_KEYS.ORDER_ID, currentOrder.id);
 
                    // Map cart-scoped tax summary to order ID
                    try {
                        if (cartId) {
                            const storedTax = sessionStorage.getItem(`${TAX_STORAGE_KEYS.CHECKOUT_TAX_SUMMARY_PREFIX}${cartId}`);
                            if (storedTax) {
                                sessionStorage.setItem(`${TAX_STORAGE_KEYS.CHECKOUT_TAX_SUMMARY_PREFIX}${currentOrder.id}`, storedTax);
                                sessionStorage.removeItem(`${TAX_STORAGE_KEYS.CHECKOUT_TAX_SUMMARY_PREFIX}${cartId}`);
                            }
                        }
                    } catch (e) {
                        console.error('Failed to map tax summary to order ID:', e);
                    }
 
                    sessionStorage.removeItem(ORDER_STORAGE_KEYS.CART_ID);
                    localStorage.removeItem(ORDER_STORAGE_KEYS.CART_ID);
                    sessionStorage.removeItem(ORDER_STORAGE_KEYS.PRODUCT_STORE_MAP);
 
                    // Robustly clear the store-scoped cart IDs to prevent stale carts after successful checkout
                    try {
                        const storeKey = localStorage.getItem(ORDER_STORAGE_KEYS.ACTIVE_STORE_KEY);
                        if (storeKey) {
                            const scopedCartKey = `${ORDER_STORAGE_KEYS.CART_ID}_${storeKey}`;
                            sessionStorage.removeItem(scopedCartKey);
                            localStorage.removeItem(scopedCartKey);
                        }
                    } catch (e) {
                        console.error('Failed to clear scoped cart ID:', e);
                    }
 
                    import('../../cart/utils/cartUtil').then(({ emitCartUpdate }) => {
                        if (emitCartUpdate) emitCartUpdate(null);
                    });
                }
            } catch (err) {
                console.error('Failed to create order from payment intent:', err);
                setError('Failed to create order');
            } finally {
                setLoading(false);
            }
        };
 
        createOrder();
    }, [paymentIntentId, invoiceOrderId]);
 
    return { loading, error, orderId };
};
