/**
 * Pure helpers for checkout order summary display.
 * No API calls; same inputs/outputs as previous inline logic.
 */
import { calculateDiscountPercentage } from '../../../shared/utils/format';
import { CT_STATES, CT_VALUE_TYPES, CT_TARGET_TYPES, UI_SYMBOLS, CHECKOUT_WARNINGS } from '../constants/checkout.constants';
import { Cart, AppliedDiscount } from '../types/checkout.types';
 
 
 
/**
 * Build applied discounts array for OrderSummary from cart and totals.
 * @param {Cart} cart - Cart with discountCodes
 * @param {number} discountAmount - Total discount in currency units (e.g. dollars)
 * @param {number} subtotalBeforeDiscount - Subtotal before discount in same units
 * @returns {AppliedDiscount[]}
 */
export function getAppliedDiscountsForSummary(
  cart: Cart | null,
  discountAmount: number,
  subtotalBeforeDiscount: number
): AppliedDiscount[] {
  if (!cart?.discountCodes) return [];
  const discountPercentage = calculateDiscountPercentage(discountAmount, subtotalBeforeDiscount);
  return cart.discountCodes
    .filter((dc) => {
      const stateValid = dc.state === CT_STATES.MATCHES_CART;
      const objPresent = !!dc.discountCode?.obj;
      const isCodeActive = dc.discountCode?.obj?.isActive !== false;
 
      // Check if underlying cart discounts are active
      const cartDiscounts = dc.discountCode?.obj?.cartDiscounts || [];
      const hasActiveCartDiscount = cartDiscounts.length === 0 ||
        cartDiscounts.some(cd => cd.obj?.isActive !== false);
 
      return stateValid && objPresent && isCodeActive && hasActiveCartDiscount;
    })
    .map((dc) => {
      const cartDiscountElement = dc.discountCode?.obj?.cartDiscounts?.[0];
      const cartDiscount = cartDiscountElement?.obj;
      const discountType = cartDiscount?.value?.type;
      
      if (discountType && !([CT_VALUE_TYPES.RELATIVE, CT_VALUE_TYPES.ABSOLUTE] as string[]).includes(discountType)) {
        console.warn(CHECKOUT_WARNINGS.UNKNOWN_DISCOUNT_TYPE(discountType));
      }
      // Relative (percentage) discount
      const nominalPercentage =
        discountType === CT_VALUE_TYPES.RELATIVE && cartDiscount?.value?.permyriad
          ? cartDiscount.value.permyriad / 100
          : null;
 
      // Absolute (fixed amount) discount — CT returns an array of money per currency
      const ctMoney = discountType === CT_VALUE_TYPES.ABSOLUTE
        ? (cartDiscount?.value?.money?.find((m) => m.currencyCode === cart?.currency) ?? cartDiscount?.value?.money?.[0] ?? null)
        : null;
      const absoluteAmount = ctMoney
        ? {
            centAmount: ctMoney.centAmount,
            currencyCode: ctMoney.currencyCode,
            fractionDigits: ctMoney.fractionDigits,
            type: (ctMoney.type as 'centPrecision' | 'highPrecision') ?? ('centPrecision' as const),
          }
        : null;
 
      // Fallback: if neither is set, use calculated % from totals (legacy path)
      const effectivePercentage = nominalPercentage ?? (absoluteAmount ? null : discountPercentage);
 
      const isProductDiscount = cartDiscountElement?.target?.type === CT_TARGET_TYPES.LINE_ITEMS;
      return {
        code: dc.discountCode?.obj?.code || dc.discountCode?.code || UI_SYMBOLS.EMPTY,
        id: dc.discountCode?.id,
        percentage: effectivePercentage,
        absoluteAmount,
        isProductDiscount,
      };
    })
    .filter(Boolean) as AppliedDiscount[];
}
