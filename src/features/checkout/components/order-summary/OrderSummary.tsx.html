import React, { useState, useEffect, useRef } from 'react';
 
import ArrowDownIcon from '../../../../assets/icons/keyboard_arrow_down.svg';
import ArrowUpIcon from '../../../../assets/icons/keyboard_arrow_up.svg';
import { appConfig } from '../../../../config/app-config';
import { useOrderSummaryBarHeight } from '../../../../shared/hooks/useOrderSummaryBarHeight';
import { formatPrice } from '../../../../shared/utils/format';
import DiscountCodeInput from '../../../discount/components/DiscountCodeInput';
import { useCheckoutDiscountContext } from '../../../discount/context/useCheckoutDiscount';
import { TaxLineItem } from '../../../tax';
import { DEFAULT_CURRENCY, DISPLAY_FALLBACK, UI_SYMBOLS, CHECKOUT_UI_TEXTS, CT_VALUE_TYPES } from '../../constants/checkout.constants';
import { DISCOUNT_BREAKDOWN_LABELS, DISCOUNT_BREAKDOWN_CONSTANTS, TOOLTIP_CONSTANTS } from '../../constants/discountBreakdown.constants';
import { ORDER_SUMMARY } from '../../constants/orderSummary.constants';
import { OrderSummaryProps, MoneyValue } from '../../types/checkout.types';
 
import DiscountBreakdownTooltip from './DiscountBreakdownTooltip';
import LineItemsPreview from './LineItemsPreview';
 
const OrderSummary: React.FC<OrderSummaryProps> = ({
    lineItems = [],
    subtotal,
    totalDiscount,
    total,
    tax,
    appliedDiscounts = [],
    discountBreakdown = [],
}) => {
    const locale = appConfig.defaults.locale;
    const [isOpen, setIsOpen] = useState(false);
    
    const scrollRef = useRef<HTMLDivElement>(null);
    const mobileScrollRef = useRef<HTMLDivElement>(null);
    const mobileAsideRef = useOrderSummaryBarHeight();
    const [isScrolling, setIsScrolling] = useState(false);
    const [isMobileScrolling, setIsMobileScrolling] = useState(false);
 
    const {
        handleApplyDiscount,
        handleRemoveDiscount,
        discountError,
        applyingDiscount,
        setDiscountError,
    } = useCheckoutDiscountContext();
 
    useEffect(() => {
        const el = scrollRef.current;
        if (!el) return;
        const observer = new ResizeObserver(() => {
            setIsScrolling(el.scrollHeight > el.clientHeight);
        });
        observer.observe(el);
        setIsScrolling(el.scrollHeight > el.clientHeight);
        return () => observer.disconnect();
    }, [lineItems]);
 
    useEffect(() => {
        const el = mobileScrollRef.current;
        if (!el) return;
        const observer = new ResizeObserver(() => {
            setIsMobileScrolling(el.scrollHeight > el.clientHeight);
        });
        observer.observe(el);
        setIsMobileScrolling(el.scrollHeight > el.clientHeight);
        return () => observer.disconnect();
    }, [lineItems, isOpen]);
 
    useEffect(() => {
        const mql = window.matchMedia('(min-width: 1200px)');
 
        const syncScrollLock = () => {
            if (isOpen && !mql.matches) {
                document.body.style.overflow = 'hidden';
            } else {
                document.body.style.overflow = '';
            }
        };
 
        syncScrollLock();
        mql.addEventListener('change', syncScrollLock);
 
        return () => {
            mql.removeEventListener('change', syncScrollLock);
            document.body.style.overflow = '';
        };
    }, [isOpen]);
 
    const format = (priceObj: MoneyValue) => {
        if (!priceObj || !priceObj.currencyCode) return DISPLAY_FALLBACK;
        return formatPrice(priceObj, locale);
    };
 
    const currencyCode = total?.currencyCode || DEFAULT_CURRENCY;
    const displayedDiscount = discountBreakdown.length > 0
        ? {
            centAmount: discountBreakdown.reduce((sum, item) => sum + item.centAmount, 0),
            currencyCode,
            type: totalDiscount?.type || CT_VALUE_TYPES.CENT_PRECISION,
            fractionDigits: totalDiscount?.fractionDigits || TOOLTIP_CONSTANTS.FRACTION_DIGITS,
          }
        : totalDiscount;
 
    const hasDiscount = (displayedDiscount?.centAmount ?? 0) > 0;
 
    const catalogSavingsCents = discountBreakdown.reduce(
        (sum, item) => item.label !== DISCOUNT_BREAKDOWN_LABELS.PROMO_CODE ? sum + item.centAmount : sum,
        0
    );
 
    const displayedSubtotal = catalogSavingsCents > 0
        ? {
            ...subtotal,
            centAmount: subtotal.centAmount + catalogSavingsCents,
          }
        : subtotal;
 
    const hasTaxData = Boolean(tax && Array.isArray(tax.aggregatedTaxLines) && tax.aggregatedTaxLines.length > 0);
 
    let finalTotalDisplay = format(total);
    if (tax?.grandTotal) {
        finalTotalDisplay = formatPrice(
            { centAmount: Math.round(Number(tax.grandTotal) * DISCOUNT_BREAKDOWN_CONSTANTS.CENTS_CONVERSION_FACTOR), currencyCode },
            locale
        );
    }
 
    const renderWhiteBoxContent = () => (
        <>
            <div className="flex flex-col w-full">
                {/* Discount form section */}
                <div className="flex flex-col gap-[11px] w-full">
                    {!hasTaxData && appliedDiscounts.length === 0 && (
                        <span className="m-0 font-noto not-italic font-medium text-[14px] leading-[14px] text-[#3B485F]">
                            {ORDER_SUMMARY.PROMOTION_CODE}
                        </span>
                    )}
 
                    {(!hasTaxData || appliedDiscounts.length > 0) && (
                        <DiscountCodeInput
                            activeCodes={appliedDiscounts.map((d) => {
                                let discountLabel: string;
                                if (d.percentage != null) {
                                    discountLabel = ORDER_SUMMARY.DISCOUNT_LABEL(d.percentage);
                                } else if (d.absoluteAmount != null) {
                                    try {
                                        const formattedAmount = format(d.absoluteAmount);
                                        if (formattedAmount) {
                                            discountLabel = ORDER_SUMMARY.DISCOUNT_AMOUNT_LABEL(formattedAmount);
                                        } else {
                                            discountLabel = ORDER_SUMMARY.DISCOUNT_APPLIED;
                                        }
                                    } catch (e) {
                                        console.error(ORDER_SUMMARY.FORMAT_ERROR, e);
                                        discountLabel = ORDER_SUMMARY.DISCOUNT_APPLIED;
                                    }
                                } else {
                                    discountLabel = ORDER_SUMMARY.DISCOUNT_APPLIED;
                                }
                                return {
                                    code: d.code,
                                    discount: discountLabel,
                                    onRemove: hasTaxData
                                        ? undefined
                                        : () => handleRemoveDiscount(d.id || UI_SYMBOLS.EMPTY),
                                };
                            })}
                            onApply={handleApplyDiscount}
                            showActiveCodes={true}
                            error={discountError}
                            isLoading={applyingDiscount}
                            onClearError={() => setDiscountError(null)}
                            className="w-full"
                            buttonClassName="min-w-[66px] shrink-0 whitespace-nowrap"
                        />
                    )}
                </div>
 
                {/* Totals section — 16px below discount form */}
                <div className="flex flex-col mt-[16px] w-full">
                    {/* Total Discount */}
                    {hasDiscount && (
                        <div className="flex justify-between items-center text-[14px] font-normal leading-[24px] text-[#002B66]/70 font-['Noto_Sans']">
                            <span>{ORDER_SUMMARY.TOTAL_DISCOUNT_LABEL}</span>
                            <span className="font-normal text-[#002B66]/70 font-['Noto_Sans'] flex items-center gap-[4px] text-right">
                                {discountBreakdown.length > 0 ? (
                                    <>
                                        <DiscountBreakdownTooltip
                                            items={discountBreakdown}
                                            currencyCode={currencyCode}
                                        />
                                        <span>-{format(displayedDiscount)}</span>
                                    </>
                                ) : (
                                    `-${format(displayedDiscount)}`
                                )}
                            </span>
                        </div>
                    )}
 
                    {/* Sub Total — 4px below Total Discount */}
                    <div className={`flex justify-between items-center text-[14px] font-normal leading-[24px] text-[#002B66B2] font-['Noto_Sans'] h-[24px] ${hasDiscount ? 'mt-[4px]' : ''}`}>
                        <span>{ORDER_SUMMARY.SUB_TOTAL_LABEL}</span>
                        <span>{format(displayedSubtotal)}</span>
                    </div>
 
                    <TaxLineItem
                        taxLines={tax?.aggregatedTaxLines ?? []}
                        currencyCode={currencyCode}
                        locale={locale}
                    />
 
                    {/* Total — 16px below Sub Total */}
                    <div className="mt-[16px]">
                        <div className="flex justify-between items-center border-gray-100 h-[26px]">
                            <span className="font-noto font-semibold text-[18px] leading-[26px] text-[#002B66]">{ORDER_SUMMARY.TOTAL}</span>
                            <span className="font-noto font-semibold text-[18px] leading-[26px] text-[#002B66] text-right">{finalTotalDisplay}</span>
                        </div>
                        {!hasTaxData && (
                            <div className="text-right h-[24px] text-[14px] mt-1 leading-[24px] font-noto font-normal text-[#002B66B2]">
                                {CHECKOUT_UI_TEXTS.TAX_BREAKDOWN_MESSAGE}
                            </div>
                        )}
                    </div>
                </div>
            </div>
        </>
    );
 
    return (
        <>
            {/* DESKTOP VIEW */}
            <aside className="hidden desktop:flex flex-col w-[410px] max-h-[calc(100vh-200px)]">
                {/* BLUE COMPONENT */}
                <div className="flex flex-col w-[410px] flex-1 min-h-[150px] pt-[24px] px-[24px] pb-[5px] bg-[linear-gradient(145.45deg,#0A51C2_0%,#001E33_95.67%)] rounded-t-[6px] shadow-[0_2px_8px_0_rgba(16,24,40,0.08),0_1px_3px_0_rgba(16,24,40,0.04)] overflow-hidden">
                    <div className="w-full flex items-center mb-[20px] shrink-0">
                        <h3 className="font-sans text-[14px] leading-[20px] font-semibold text-white m-0">
                            {ORDER_SUMMARY.HEADING}
                        </h3>
                    </div>
                    <div
                        ref={scrollRef}
                        className={`flex flex-col overflow-y-auto flex-1 min-h-[80px] pb-[20px] ${isScrolling ? 'pr-[20px]' : ''} [&::-webkit-scrollbar]:w-[16px] [&::-webkit-scrollbar-track]:bg-transparent [&::-webkit-scrollbar-thumb]:border-[4px] [&::-webkit-scrollbar-thumb]:border-solid [&::-webkit-scrollbar-thumb]:border-transparent [&::-webkit-scrollbar-thumb]:bg-clip-padding [&::-webkit-scrollbar-thumb]:bg-white/20 [&::-webkit-scrollbar-thumb]:rounded-full`}
                    >
                        <LineItemsPreview lineItems={lineItems} showTaxBreakdown={hasTaxData} />
                    </div>
                </div>
 
                {/* WHITE COMPONENT */}
                <div className="w-[410px] p-[24px] bg-white rounded-[6px] shadow-[0_2px_8px_0_rgba(16,24,40,0.08),0_1px_3px_0_rgba(16,24,40,0.04)] -mt-[5px] relative z-10 flex flex-col shrink-0">
                    {renderWhiteBoxContent()}
                </div>
            </aside>
 
            <aside
                ref={mobileAsideRef}
                className="js-mobile-order-summary desktop:hidden fixed bottom-0 left-0 right-0 z-50 flex flex-col shadow-[0_-4px_20px_rgba(0,0,0,0.1)] rounded-t-[12px] tablet:rounded-t-none overflow-visible"
            >
                {/* BLUE GRADIENT WRAPPER */}
                <div
                    className={`w-full bg-[linear-gradient(145.45deg,#0A51C2_0%,#001E33_95.67%)] rounded-t-[12px] tablet:rounded-tl-[6px] tablet:rounded-tr-[6px] flex flex-col pt-[16px] px-[16px] ${isOpen ? 'pb-[5px]' : 'pb-[21px]'}`}
                >
                    <div className={`w-full flex flex-col`}>
                        <div className="w-full h-[20px] flex items-center justify-between">
                            <span className="font-sans text-[16px] tablet:text-[14px] leading-[20px] font-semibold text-white">
                                {isOpen ? ORDER_SUMMARY.HEADING : `${ORDER_SUMMARY.HEADING} — ${finalTotalDisplay}`}
                            </span>
                            <button
                                type="button"
                                aria-expanded={isOpen}
                                aria-controls="checkout-order-summary-content"
                                aria-label={isOpen ? ORDER_SUMMARY.COLLAPSE_LABEL : ORDER_SUMMARY.EXPAND_LABEL}
                                className="flex items-center justify-center w-[20px] h-[20px] border-none bg-transparent p-0 outline-none cursor-pointer focus-visible:outline-2 focus-visible:outline-offset-2 focus-visible:outline-white"
                                onClick={() => setIsOpen((prev) => !prev)}
                            >
                                <img
                                    src={isOpen ? ArrowUpIcon : ArrowDownIcon}
                                    className="w-[20px] h-[20px] shrink-0"
                                    alt=""
                                    aria-hidden="true"
                                />
                            </button>
                        </div>
                        <div
                            id="checkout-order-summary-content"
                            role="region"
                            aria-label={ORDER_SUMMARY.HEADING}
                        >
                            {isOpen && (
                                <div
                                    ref={mobileScrollRef}
                                    className={`w-full px-0 mt-[20px] pb-[20px] min-h-[120px] max-h-[calc(100vh-498px)] tablet:max-h-[calc(100vh-452px)] overflow-y-auto ${isMobileScrolling ? 'pr-[20px]' : ''} [&::-webkit-scrollbar]:w-[16px] [&::-webkit-scrollbar-track]:bg-transparent [&::-webkit-scrollbar-thumb]:border-[4px] [&::-webkit-scrollbar-thumb]:border-solid [&::-webkit-scrollbar-thumb]:border-transparent [&::-webkit-scrollbar-thumb]:bg-clip-padding [&::-webkit-scrollbar-thumb]:bg-white/20 [&::-webkit-scrollbar-thumb]:rounded-full`}
                                >
                                    <LineItemsPreview lineItems={lineItems} showTaxBreakdown={hasTaxData} />
                                </div>
                            )}
                        </div>
                    </div>
                </div>
 
                {/* WHITE CARD */}
                {isOpen && (
                    <div className="w-full bg-white rounded-t-[6px] tablet:rounded-tl-[6px] tablet:rounded-tr-[6px] -mt-[5px] tablet:-mt-[5px] relative z-10 pt-[17px] tablet:pt-[17px] px-[16px] tablet:px-[16px] pb-[24px] tablet:pb-[24px] flex flex-col">
                        <div className="w-full flex flex-col gap-[20px]">
                            {renderWhiteBoxContent()}
                        </div>
                    </div>
                )}
            </aside>
        </>
    );
};
 
export default OrderSummary;
