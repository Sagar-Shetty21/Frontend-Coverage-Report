/**
 * Route config. Same paths and components as before; no behavior change.
 * Pages imported from feature entry points where available.
 * Each route is wrapped in ErrorBoundary so a crash on one page
 * does not take down the entire app.
 */
import React from 'react';
import { Navigate, Route, Routes } from 'react-router-dom';
 
import AccountPage from '../features/auth/account-page';
import PingCallback from '../features/auth/ping-callback';
import { CartPage } from '../features/cart';
import { CheckoutPage } from '../features/checkout';
import HomePage from '../features/home/components/HomePage';
import EventPrivacyPolicyPage from '../features/legal/EventPrivacyPolicyPage';
import InvoicePage from '../features/order/components/InvoicePage';
import OrderPage from '../features/order/components/OrderPage';
import { RegisterPassPage } from '../features/register-pass';
import { SharedBasketPage } from '../features/shared-basket';
import StoreLayout from '../features/store/components/StoreLayout';
import NotFound from '../shared/components/NotFound';
import ProtectedRoute from '../shared/components/ProtectedRoute';
import { getActiveKeys } from '../shared/utils/storage';
 
import ErrorBoundary from './layout/ErrorBoundary';
import EventGuard from './layout/EventGuard';
 
/** Wrap a page element in an ErrorBoundary */
const safe = (element: React.ReactElement): React.ReactElement => <ErrorBoundary>{element}</ErrorBoundary>;
 
/**
 * Redirects to the last visited store/event from localStorage,
 * falling back to appConfig defaults when no cache exists.
 */
const CachedRedirect = () => {
  const { storeKey, eventKey } = getActiveKeys();
  return <Navigate to={`/${storeKey}/${eventKey}`} replace />;
};
 
export default function AppRoutes() {
  return (
    <Routes>
      <Route path="/ping-callback" element={safe(<PingCallback />)} />
      <Route path="/shared-basket/:shareToken" element={<ErrorBoundary fallback={<NotFound />}><SharedBasketPage /></ErrorBoundary>} />
      <Route path="/shared-basket" element={<NotFound />} />
 
      <Route path="/:storeKey" element={<StoreLayout />}>
        {/* Bare store URL (no eventKey) is invalid — show 404 */}
        <Route index element={<NotFound />} />
        {/* Store-Level Routes */}
        <Route path="shared-basket/:shareToken" element={<ErrorBoundary fallback={<NotFound />}><SharedBasketPage /></ErrorBoundary>} />
        <Route path="shared-basket" element={<NotFound />} />
        <Route path="cart" element={safe(<CartPage />)} />
        <Route path="checkout" element={
          <ProtectedRoute>
            {safe(<CheckoutPage />)}
          </ProtectedRoute>
        } />
        <Route path="invoice-confirmation" element={safe(<InvoicePage />)} />
        <Route path="register-pass" element={safe(<RegisterPassPage />)} />
 
        <Route path="account" element={safe(<AccountPage />)} />
        <Route path="order" element={
          <ProtectedRoute>
            {safe(<OrderPage />)}
          </ProtectedRoute>
        } />
 
        {/* Event-Level Routes — guarded by legalEntityCode match */}
        <Route path=":eventKey" element={<EventGuard />}>
          <Route index element={safe(<HomePage />)} />
          <Route path="account" element={safe(<AccountPage />)} />
          <Route path="event-privacy-policy" element={safe(<EventPrivacyPolicyPage />)} />
        </Route>
      </Route>
 
      <Route path="/cart" element={<NotFound />} />
      <Route path="/checkout" element={<NotFound />} />
      <Route path="/account" element={<NotFound />} />
      <Route path="/order" element={<NotFound />} />
      <Route path="/store" element={<NotFound />} />
 
      <Route path="/" element={<CachedRedirect />} />
      <Route path="*" element={<NotFound />} />
    </Routes>
  );
}
