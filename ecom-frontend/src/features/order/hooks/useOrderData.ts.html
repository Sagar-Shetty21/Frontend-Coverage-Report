const getLocText = (val: unknown): string => {
    if (!val) return '';
    if (typeof val === 'string') return val;
    const v = val as Record<string, string>;
    return v[LOCALE_EN] || (Object.values(v)[0] as string) || '';
};
 
const formatDate = (
    dateString: string | undefined | null,
    options: Intl.DateTimeFormatOptions = ORDER_DATE_OPTIONS
): string => {
    if (!dateString) return '';
    try {
        return new Date(dateString).toLocaleDateString(LOCALE_EN, options);
    } catch (err) {
        console.error('Failed to format date:', err);
        return String(dateString);
    }
};
 
/**
 * Hook that fetches order data for card-payment confirmation.
 * Handles Stripe session/payment-intent flows and stored order IDs.
 *
 * paymentIntentId is sourced strictly from React Router navigate state (location.state).
 * This ensures the context is cleared on any browser reload — the correct security behaviour.
 */
 
export const useOrderData = (): UseOrderDataReturn => {
    const eventContext = useContext(EventContext);
    const customerGroups = useMemo(() => eventContext?.customerGroups ?? [], [eventContext?.customerGroups]);
    const [order, setOrder] = useState<OrderWithPayment | null>(null);
    const [products, setProducts] = useState<DisplayProduct[]>([]);
    const [customerEmail, setCustomerEmail] = useState<string | null>(null);
    const [loading, setLoading] = useState(true);
    const [error, setError] = useState<string | null>(null);
    const [nextSteps, setNextSteps] = useState<NextStep[]>([]);
    const [hotelMapInfo, setHotelMapInfo] = useState<HotelMapInfo | null>(null);
 
    const storedOrderId = sessionStorage.getItem(ORDER_STORAGE_KEYS.ORDER_ID);
 
    const processedRef = useRef(false);
    const lastParamsRef = useRef<string | null>(null);
 
    useEffect(() => {
        const paramsKey = `${storedOrderId ?? ''}_${customerGroups.length}`;
        if (lastParamsRef.current !== paramsKey) {
            lastParamsRef.current = paramsKey;
            processedRef.current = false;
        }
 
        const fetchOrderData = async () => {
            if (processedRef.current) return;
            processedRef.current = true;
 
            const startTime = Date.now();
 
            try {
                let currentOrder: OrderWithPayment | null = null;
 
                if (storedOrderId) {
                    currentOrder = (await getOrderById(storedOrderId)) as OrderWithPayment | null;
                }
 
                if (currentOrder) {
                    setOrder(currentOrder);
                    setCustomerEmail(currentOrder.customerEmail ?? '');
 
                    const { processedProducts, configProductId, hotelMapInfo: hotelInfo } = await processLineItems(currentOrder, customerGroups);
                    setProducts(processedProducts);
                    if (hotelInfo) {
                        setHotelMapInfo(hotelInfo);
                    }
 
                    if (configProductId) {
                        // Amber country orders use their own next-steps attribute
                        const isAmber = (currentOrder as unknown as { custom?: { fields?: Record<string, unknown> } })
                            ?.custom?.fields?.[ORDER_FIELD_KEYS.IS_AMBER_COUNTRY] === true;
                        const nextStepsKey = isAmber
                            ? ORDER_ATTRIBUTE_KEYS.AMBER_NEXT_STEPS
                            : ORDER_ATTRIBUTE_KEYS.CARD_NEXT_STEPS;
                        const steps = await fetchNextSteps(configProductId, nextStepsKey);
                        if (steps.length > 0) {
                            setNextSteps(steps);
                        }
                    }
                }
            } catch (err) {
                console.error('Failed to load order data:', err);
                setError(ORDER_TEXTS.ERRORS.FETCH_FAILED);
            } finally {
                const elapsedTime = Date.now() - startTime;
                const remainingTime = Math.max(0, MIN_LOAD_TIME - elapsedTime);
                setTimeout(() => setLoading(false), remainingTime);
            }
        };
 
        fetchOrderData();
    }, [storedOrderId, customerGroups]);
 
    return { order, products, customerEmail, nextSteps, hotelMapInfo, loading, error };
};

/**
 * Processes line items from an order into DisplayProduct[] for confirmation UI.
 */
export const processLineItems = async (
    currentOrder: Order & { lineItems?: OrderLineItem[] },
    customerGroups: CustomerGroup[] = []
): Promise<{ processedProducts: DisplayProduct[]; configProductId: string | null; hotelMapInfo: HotelMapInfo | null }> => {
    const associationGroups: Record<string, AssociationGroup> = {};
    const lineItems = currentOrder.lineItems ?? [];
 
    lineItems.forEach((item: OrderLineItem) => {
        const assocKey = item.custom?.fields?.[ORDER_FIELD_KEYS.ASSOCIATION_KEY] as string | undefined;
        if (assocKey) {
            if (!associationGroups[assocKey]) {
                associationGroups[assocKey] = { passes: [], addons: [] };
            }
            const productTypeId = item.productType?.id;
            if (productTypeId === PASS_PRODUCT_TYPE_ID) {
                associationGroups[assocKey].passes.push(item);
            } else if (productTypeId === ADDON_PRODUCT_TYPE_ID) {
                associationGroups[assocKey].addons.push(item);
            }
        }
    });
 
    let configurationProductId: string | null = null;
    let hotelMapInfo: HotelMapInfo | null = null;
    const productsList = await Promise.all(
        Object.entries(associationGroups).map(async ([, group]) => {
            const firstPass = group.passes[0];
            if (!firstPass) return null;
 
            const productData = await getProductById(firstPass.productId);
            const totalPassQuantity = group.passes.reduce((sum, pass) => sum + (pass.quantity ?? 1), 0);
            const totalPassPrice = group.passes.reduce((sum, pass) => {
                const unitPrice = pass.price?.discounted?.value?.centAmount ?? pass.price?.value?.centAmount ?? 0;
                return sum + unitPrice * (pass.quantity ?? 1);
            }, 0);
 
            let unitPassPrice = 0;
            let unitOriginalPrice = 0;
 
            if (firstPass.price) {
                if (firstPass.price.discounted) {
                    unitPassPrice = firstPass.price.discounted.value?.centAmount ?? 0;
                    unitOriginalPrice = firstPass.price.value?.centAmount ?? 0;
                } else {
                    unitPassPrice = firstPass.price.value?.centAmount ?? 0;
                    unitOriginalPrice = 0;
                }
            }
 
            let productName: string = ORDER_TEXTS.UNKNOWN_PRODUCT;
            let productLocation: string | null = null;
            let productDate: string | null = null;
            let startDate: string | null = null;
            let endDate: string | null = null;
            let categoryName = '';
            let categoryStartDate: string | null = null;
            let categoryEndDate: string | null = null;
            let categoryLocation: string | null = null;
 
            if (productData) {
                interface CommercetoolsProductPrice {
                    value: { centAmount: number; currencyCode: string };
                    discounted?: { value: { centAmount: number } };
                    customerGroup?: { id: string; obj?: { id: string; key?: string } };
                }
                interface CommercetoolsProduct {
                    name?: Record<string, string>;
                    attributes?: Array<{ name: string; value?: unknown }>;
                    masterVariant?: {
                        attributes?: Array<{ name: string; value?: unknown }>;
                        prices?: CommercetoolsProductPrice[];
                    };
                    variants?: Array<{
                        id: number;
                        attributes?: Array<{ name: string; value?: unknown }>;
                        prices?: CommercetoolsProductPrice[];
                    }>;
                    categories?: Array<{ id: string }>;
                }
                const productRec = productData as unknown as {
                    masterData?: {
                        current?: CommercetoolsProduct;
                        staged?: CommercetoolsProduct;
                    }
                };
                const currentData = (productRec.masterData?.current || productRec.masterData?.staged) as CommercetoolsProduct | undefined;
 
                const nameObj = currentData?.name as Record<string, string> | undefined;
                if (nameObj) {
                    productName = nameObj[LOCALE_EN] || (Object.values(nameObj)[0] as string);
                }
 
                if (unitOriginalPrice === 0 || unitOriginalPrice <= unitPassPrice) {
                    const rateName = (firstPass.custom?.fields?.[ORDER_FIELD_KEYS.DELEGATE_RATE] as string) || '';
                    const rateId = customerGroups.find((g) => {
                        const name = g.name;
                        if (!name) return false;
                        if (typeof name === 'string') return name === rateName;
                        const groupName = name[LOCALE_EN_US] || name[LOCALE_EN] || name[LOCALE_EN_GB] || (Object.values(name)[0] as string) || '';
                        return groupName === rateName;
                    })?.id ?? null;
 
                    const savings = getPassCatalogSavings(productData as unknown as ProductWithAddOns, rateId, customerGroups);
                    let unitSavedCents = savings.unitSavedCents;
                    let isCatalogDiscountActive = savings.isCatalogDiscountActive;
 
                    // Local resilient fallback: if getPassCatalogSavings returned 0 because discountObj
                    // reference is not expanded in the API fetch, resolve it using local price object fields
                    if (!isCatalogDiscountActive && currentData) {
                        const prices = currentData.masterVariant?.prices || [];
                        const hasCustomerGroupPrice = prices.some((p) => p.customerGroup);
                        let priceObj = prices[0];
                        if (hasCustomerGroupPrice && rateId) {
                            priceObj = prices.find((p) => p.customerGroup && p.customerGroup.id === rateId) ?? priceObj;
                            if (!priceObj?.discounted && customerGroups) {
                                const selectedGroup = customerGroups.find((g) => g.id === rateId);
                                if (selectedGroup?.key) {
                                    priceObj = prices.find((p) => p.customerGroup?.obj?.key === selectedGroup.key) ?? priceObj;
                                }
                            }
                        }
 
                        if (priceObj?.discounted?.value?.centAmount) {
                            const origAmount = priceObj.value?.centAmount ?? 0;
                            const discAmount = priceObj.discounted.value.centAmount ?? 0;
                            const saved = origAmount - discAmount;
                            if (saved > 0) {
                                unitSavedCents = saved;
                                isCatalogDiscountActive = true;
                            }
                        }
                    }
 
                    if (isCatalogDiscountActive && unitSavedCents > 0) {
                        unitOriginalPrice = unitPassPrice + unitSavedCents;
                    } else {
                        unitOriginalPrice = unitPassPrice;
                    }
                }
 
                const categories = currentData?.categories as Array<{ id: string }> | undefined;
                if (categories?.length) {
                    const categoryData = await getCategoryById(categories[0].id);
                    if (categoryData) {
                        categoryName = categoryData.name?.[LOCALE_EN] || (categoryData.name ? Object.values(categoryData.name)[0] : '') || '';
                        categoryStartDate = (categoryData.custom?.fields?.startDate as string) ?? null;
                        categoryEndDate = (categoryData.custom?.fields?.endDate as string) ?? null;
                        
                        const venueName = (categoryData.custom?.fields?.eventVenueName as string) || '';
                        const state = (categoryData.custom?.fields?.eventStateProvince as string) || '';
                        const country = (categoryData.custom?.fields?.eventCountryCode as string) || '';
                        
                        const locationParts = [venueName, state, country].filter(Boolean);
                        categoryLocation = locationParts.join(', ') || null;
                    }
 
                    const categoryFields = (categoryData as unknown as Record<string, unknown>)?.custom as
                        | { fields?: Record<string, unknown> }
                        | undefined;
 
                    // ── Hotel map: extract once from first resolved category ──
                    if (!hotelMapInfo && categoryFields?.fields) {
                        const status = categoryFields.fields[ORDER_FIELD_KEYS.HOTEL_MAP_STATUS];
                        const url = categoryFields.fields[ORDER_FIELD_KEYS.HOTEL_MAP_URL];
                        if (typeof status === 'string' && typeof url === 'string') {
                            hotelMapInfo = { status, url };
                        }
                    }
 
                    const confProduct = categoryFields as { fields?: { [ORDER_FIELD_KEYS.CONFIG_PRODUCT]?: { id?: string } } } | undefined;
                    if (!configurationProductId && confProduct?.fields?.[ORDER_FIELD_KEYS.CONFIG_PRODUCT]?.id) {
                        configurationProductId = confProduct.fields[ORDER_FIELD_KEYS.CONFIG_PRODUCT]!.id!;
                    }
 
                    const configurationProductRef = (categoryData as { custom?: { fields?: { [ORDER_FIELD_KEYS.CONFIG_PRODUCT]?: { id?: string; obj?: unknown } } } })?.custom?.fields?.[ORDER_FIELD_KEYS.CONFIG_PRODUCT];
                    if (configurationProductRef) {
                        let configProduct = configurationProductRef.obj as { masterData?: { current?: { masterVariant?: { attributes?: Array<{ name: string; value?: unknown }> } }; staged?: { masterVariant?: { attributes?: Array<{ name: string; value?: unknown }> } } } } | null;
                        if (!configProduct) {
                            configProduct = await getProductById(configurationProductRef.id!) as unknown as { masterData?: { current?: { masterVariant?: { attributes?: Array<{ name: string; value?: unknown }> } }; staged?: { masterVariant?: { attributes?: Array<{ name: string; value?: unknown }> } } } };
                        }
                        if (configProduct) {
                            const configAttributes =
                                configProduct.masterData?.current?.masterVariant?.attributes ||
                                configProduct.masterData?.staged?.masterVariant?.attributes ||
                                [];
                            const eventVenue = (configAttributes as Array<{ name: string; value: unknown }>).find((a) => a.name === ORDER_FIELD_KEYS.EVENT_VENUE)?.value as string | undefined;
                            const eventLocation = (configAttributes as Array<{ name: string; value: unknown }>).find((a) => a.name === ORDER_FIELD_KEYS.EVENT_LOCATION)?.value as string | undefined;
                            if (eventVenue && eventLocation) productLocation = `${eventVenue}, ${eventLocation}`;
                            else productLocation = eventVenue || eventLocation || null;
                        }
                    } else {
                        const venue = ((categoryData as unknown as Record<string, unknown>)?.custom as { fields?: { [ORDER_FIELD_KEYS.VENUE]?: string } })?.fields?.[ORDER_FIELD_KEYS.VENUE];
                        if (venue) productLocation = venue;
                    }
                }
 
                let attributes: Array<{ name: string; value?: unknown }> =
                    currentData?.masterVariant?.attributes || currentData?.attributes || [];
                const allVars = [currentData?.masterVariant, ...(currentData?.variants ?? [])];
                const match = allVars.find((v) => (v as { id?: number })?.id === firstPass.variant?.id);
                if (match) attributes = (match as { attributes?: Array<{ name: string; value?: unknown }> }).attributes ?? [];
                if (firstPass.variant?.attributes) attributes = firstPass.variant.attributes;
 
                startDate = (attributes.find((a) => a.name === ORDER_FIELD_KEYS.START_DATE)?.value as string) ?? null;
                endDate = (attributes.find((a) => a.name === ORDER_FIELD_KEYS.END_DATE)?.value as string) ?? null;
                if (startDate && endDate) productDate = `${formatDate(startDate)} – ${formatDate(endDate)}`;
                else if (startDate) productDate = formatDate(startDate);
            }
 
            const addons = group.addons.map((addon) => {
                const nameObj = (addon as unknown as LineItem).name;
                const name = nameObj ? (nameObj[LOCALE_EN] || Object.values(nameObj)[0] || ORDER_TEXTS.UNKNOWN_ADDON) : ORDER_TEXTS.UNKNOWN_ADDON;
                const price = addon.price?.discounted?.value?.centAmount ?? addon.price?.value?.centAmount ?? 0;
                const totalPrice = price * (addon.quantity ?? 1);
                return {
                    id: addon.id,
                    name,
                    price,
                    quantity: addon.quantity ?? 1,
                    totalPrice,
                };
            });
 
            return {
                id: firstPass.id,
                name: productName,
                price: unitPassPrice,
                originalPrice: unitOriginalPrice > unitPassPrice ? unitOriginalPrice : null,
                totalPrice: totalPassPrice,
                date: productDate,
                startDate,
                endDate,
                location: productLocation,
                addOnsCount: group.addons.length,
                addons,
                quantity: totalPassQuantity,
                categoryName,
                categoryStartDate,
                categoryEndDate,
                categoryLocation,
                delegateRate: (firstPass.custom?.fields?.[ORDER_FIELD_KEYS.DELEGATE_RATE] as string) || null,
            } as DisplayProduct;
        })
    );
 
    return {
        processedProducts: productsList.filter((p): p is DisplayProduct => p != null),
        configProductId: configurationProductId,
        hotelMapInfo,
    };
};
 
 
/**
 * Fetches next-step items from a configuration product's attributes.
 */
export const fetchNextSteps = async (
    configProductId: string,
    attributeName: string
): Promise<NextStep[]> => {
    try {
        const product = await getProductById(configProductId);
        if (!product) return [];
 
        const current = (product as unknown as { masterData?: { current?: { masterVariant?: { attributes?: Array<{ name: string; value?: unknown }> } }; staged?: { masterVariant?: { attributes?: Array<{ name: string; value?: unknown }> } } } }).masterData?.current || (product as unknown as { masterData?: { current?: { masterVariant?: { attributes?: Array<{ name: string; value?: unknown }> } }; staged?: { masterVariant?: { attributes?: Array<{ name: string; value?: unknown }> } } } }).masterData?.staged;
        const attributes = current?.masterVariant?.attributes ?? [];
        const stepsAttr = (attributes as Array<{ name: string; value?: unknown }>).find((a) => a.name === attributeName);
        const stepsValue = stepsAttr?.value;
 
        if (stepsValue && Array.isArray(stepsValue)) {
             const formattedSteps = (stepsValue as Array<Array<{ name: string; value?: unknown }>>) .map((stepAttrs) => {
                const titleAttr = stepAttrs.find((x) => x.name === ORDER_FIELD_KEYS.STEP_TITLE);
                const textAttr = stepAttrs.find((x) => x.name === ORDER_FIELD_KEYS.STEP_DESCRIPTION);
                const iconAttr = stepAttrs.find((x) => x.name === ORDER_FIELD_KEYS.STEP_ICON || x.name === ORDER_FIELD_KEYS.STEP_ICON_ALT || x.name === ORDER_FIELD_KEYS.STEP_ICON_UNDERSCORE);
                return {
                    title: getLocText(titleAttr?.value),
                    description: getLocText(textAttr?.value),
                    icon: (iconAttr?.value as string) || NEXT_STEPS_ICONS.FILE,
                };
            });
            return formattedSteps;
        }
