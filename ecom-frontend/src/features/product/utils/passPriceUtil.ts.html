import { appConfig } from '../../../config/app-config';
import { DEFAULT_LOCALE, UI_SYMBOLS } from '../../checkout/constants/checkout.constants';
import { DISCOUNT_BREAKDOWN_CONSTANTS, PRODUCT_PRICE_CONSTANTS } from '../../checkout/constants/discountBreakdown.constants';
import type { CustomerGroup } from '../../home/types/home.types';
import { PRODUCT_KEYS } from '../constants/product.constants';
import type { ProductWithAddOns } from '../types/product.types';
 
export interface PassPriceInfo {
  displayPrice: string;
  originalPrice: string | null;
  discountName: string | null;
  savedAmount: string | null;
}
 
export const formatPrice = (price?: { value?: { centAmount?: number; currencyCode?: string } }) => {
  if (!price || !price.value || !price.value.currencyCode) return PRODUCT_PRICE_CONSTANTS.NOT_AVAILABLE;
  const amount = (price.value.centAmount || 0) / DISCOUNT_BREAKDOWN_CONSTANTS.CENTS_CONVERSION_FACTOR;
  return new Intl.NumberFormat(DEFAULT_LOCALE, {
    style: PRODUCT_PRICE_CONSTANTS.STYLE_CURRENCY,
    currency: price.value.currencyCode,
  }).format(amount);
};
 
export const getPassPriceInfo = (
  product: ProductWithAddOns,
  selectedRateId?: string | null,
  customerGroups?: CustomerGroup[]
): PassPriceInfo => {
  const locale = appConfig.defaults.locale || DEFAULT_LOCALE;
 
  // 1. Resolve Master Variant
  type MasterVariant = NonNullable<ProductWithAddOns['masterVariant']> & {
    prices?: Array<{
      customerGroup?: { id: string; obj?: { id: string; key?: string } };
      value?: { centAmount?: number; currencyCode?: string };
      discounted?: {
        value?: { centAmount?: number; currencyCode?: string };
        discount?: { obj?: { name: Record<string, string> } };
      };
    }>;
  };
 
  const masterVariant = (product?.[PRODUCT_KEYS.MASTER_VARIANT] ||
    (
      product as unknown as {
        masterData?: { current?: { masterVariant?: MasterVariant } };
      }
    )?.[PRODUCT_KEYS.MASTER_DATA]?.[PRODUCT_KEYS.CURRENT]?.[PRODUCT_KEYS.MASTER_VARIANT]) as MasterVariant | undefined;
 
  // 2. Resolve target Price Object
  const hasCustomerGroupPrice = masterVariant?.prices?.some((p) => p.customerGroup);
  let priceObj: MasterVariant['prices'] extends (infer P)[] | undefined ? P : never =
    undefined as never;
 
  if (hasCustomerGroupPrice && selectedRateId) {
    priceObj = masterVariant?.prices?.find(
      (p) => p.customerGroup && p.customerGroup.id === selectedRateId
    ) as typeof priceObj;
 
    if (!priceObj && customerGroups) {
      const selectedGroup = customerGroups.find((g) => g.id === selectedRateId);
      if (selectedGroup?.key) {
        priceObj = masterVariant?.prices?.find(
          (p) => p.customerGroup?.obj?.key === selectedGroup.key
        ) as typeof priceObj;
      }
    }
  }
 
  if (!priceObj) {
    priceObj = masterVariant?.prices?.[0] as typeof priceObj;
  }
 
  // 3. Format primary display prices
  let displayPrice: string = PRODUCT_PRICE_CONSTANTS.NOT_AVAILABLE;
  let originalPrice: string | null = null;
 
  if (priceObj) {
    if ((priceObj as { discounted?: unknown }).discounted) {
      displayPrice = formatPrice(
        (priceObj as { discounted: { value?: { centAmount?: number; currencyCode?: string } } })
          .discounted
      );
      originalPrice = formatPrice(
        priceObj as { value?: { centAmount?: number; currencyCode?: string } }
      );
    } else {
      displayPrice = formatPrice(
        priceObj as { value?: { centAmount?: number; currencyCode?: string } }
      );
    }
  }
 
  // 4. Extract specific discounts/savings
  let discountName: string | null = null;
  let savedAmount: string | null = null;
 
  type PriceWithDiscount = {
    discounted?: {
      discount?: { obj?: { name?: Record<string, string> } };
      value?: { centAmount?: number };
    };
    value?: { centAmount?: number; currencyCode?: string };
  };
 
  const priceWithDiscount = priceObj as PriceWithDiscount | undefined;
  if (priceWithDiscount?.discounted?.discount?.obj) {
    const discountObj = priceWithDiscount.discounted.discount.obj;
    discountName = (discountObj.name?.[locale] || discountObj.name?.[DEFAULT_LOCALE]) ?? null;
 
    const originalAmount = priceWithDiscount.value?.centAmount || 0;
    const discountedAmount = priceWithDiscount.discounted.value?.centAmount || 0;
    const savedCents = originalAmount - discountedAmount;
 
    if (savedCents > 0) {
      savedAmount = new Intl.NumberFormat(DEFAULT_LOCALE, {
        style: PRODUCT_PRICE_CONSTANTS.STYLE_CURRENCY,
        currency: priceWithDiscount.value?.currencyCode || PRODUCT_PRICE_CONSTANTS.DEFAULT_CURRENCY,
        minimumFractionDigits: PRODUCT_PRICE_CONSTANTS.MIN_FRACTION_DIGITS_ZERO,
        maximumFractionDigits: PRODUCT_PRICE_CONSTANTS.MAX_FRACTION_DIGITS_ZERO,
      }).format(savedCents / DISCOUNT_BREAKDOWN_CONSTANTS.CENTS_CONVERSION_FACTOR);
      savedAmount = `${UI_SYMBOLS.DASH}${savedAmount}`;
    }
  }
 
  return { displayPrice, originalPrice, discountName, savedAmount };
};
 
export interface PassCatalogSavings {
  discountName: string | null;
  unitSavedCents: number;
  /** Matches homepage ProductPassList early-bird / catalog discount detection. */
  isCatalogDiscountActive: boolean;
}
 
/**
 * Cent savings per unit for a pass catalog discount (early bird, etc.).
 * Same price resolution and formula as getPassPriceInfo / homepage PassCard.
 */
export function getPassCatalogSavings(
  product: ProductWithAddOns,
  selectedRateId?: string | null,
  customerGroups?: CustomerGroup[]
): PassCatalogSavings {
  const locale = appConfig.defaults.locale || DEFAULT_LOCALE;
 
  type MasterVariant = NonNullable<ProductWithAddOns['masterVariant']> & {
    prices?: Array<{
      customerGroup?: { id: string; obj?: { id: string; key?: string } };
      value?: { centAmount?: number; currencyCode?: string };
      discounted?: {
        value?: { centAmount?: number };
        discount?: { obj?: { name?: Record<string, string> } };
      };
    }>;
    attributes?: Array<{ name: string; value: unknown }>;
  };
 
  const masterVariant = (product?.[PRODUCT_KEYS.MASTER_VARIANT] ||
    (
      product as unknown as {
        masterData?: { current?: { masterVariant?: MasterVariant } };
      }
    )?.[PRODUCT_KEYS.MASTER_DATA]?.[PRODUCT_KEYS.CURRENT]?.[PRODUCT_KEYS.MASTER_VARIANT]) as MasterVariant | undefined;
 
  const hasCustomerGroupPrice = masterVariant?.prices?.some((p) => p.customerGroup);
  let priceObj = masterVariant?.prices?.[0];
 
  if (hasCustomerGroupPrice && selectedRateId) {
    priceObj =
      masterVariant?.prices?.find(
        (p) => p.customerGroup && p.customerGroup.id === selectedRateId
      ) ?? priceObj;
 
    if (!priceObj?.discounted && customerGroups) {
      const selectedGroup = customerGroups.find((g) => g.id === selectedRateId);
      if (selectedGroup?.key) {
        priceObj =
          masterVariant?.prices?.find(
            (p) => p.customerGroup?.obj?.key === selectedGroup.key
          ) ?? priceObj;
      }
    }
  }
 
  const marketingTag = masterVariant?.attributes?.find(
    (a) => a.name === PRODUCT_PRICE_CONSTANTS.MARKETING_TAG_LOWER || a.name === PRODUCT_PRICE_CONSTANTS.MARKETING_TAG_CAMEL
  )?.value;
 
  const discountObj = priceObj?.discounted?.discount?.obj;
  const discountName = discountObj
    ? (discountObj.name?.[locale] || discountObj.name?.[DEFAULT_LOCALE]) ?? null
    : null;
 
  const isEarlyBird =
    !!discountName ||
    (typeof marketingTag === 'string' &&
      marketingTag.toLowerCase().includes(DISCOUNT_BREAKDOWN_CONSTANTS.EARLY_BIRD_TAG));
 
  if (!priceObj?.discounted?.value?.centAmount || !discountObj) {
    return { discountName, unitSavedCents: 0, isCatalogDiscountActive: false };
  }
 
  const originalAmount = priceObj.value?.centAmount ?? 0;
  const discountedAmount = priceObj.discounted.value.centAmount ?? 0;
  const unitSavedCents = originalAmount - discountedAmount;
 
  return {
    discountName,
    unitSavedCents: unitSavedCents > 0 ? unitSavedCents : 0,
    isCatalogDiscountActive: unitSavedCents > 0 && isEarlyBird,
  };
}
