import * as React from 'react';
 
import AsteriskIcon from '../../../../assets/icons/asterisk.svg';
import CalendarIcon from '../../../../assets/icons/calendar_today.svg';
import ConfirmationIcon from '../../../../assets/icons/confirmation_number.svg';
import EditIcon from '../../../../assets/icons/edit.svg';
import PromoIcon from '../../../../assets/icons/promo.svg';
import UserAttributesIcon from '../../../../assets/icons/user_attributes.svg';
import DateRange from '../../../../shared/components/DateRange';
import QuantityControl from '../../../../shared/components/QuantityControl';
import { formatDateRange } from '../../../../shared/utils/format';
import { PassConfigurationModal } from '../../../product';
import type { RawAddOnProduct } from '../../../product/hooks/usePassProcessor';
import { CART_COPY, CART_DEFAULTS, formatDiscountMessage } from '../../constants/constants';
import { formatCurrency } from '../../utils/cartFormatting';
 
 
import RemovePassModal from './RemovePassModal';
 
export interface BasketItemProps {
  title: string;
  children?: React.ReactNode;
  addOns?: Array<{ id?: string; name: string; price?: number; totalPrice?: number }>;
  price: number;
  originalPrice?: number;
  quantity?: number;
  onQuantityChange?: (qty: number) => void;
  onRemove?: () => void;
  promoLabel?: string;
  promoValue?: string;
  currency?: string;
  locale?: string;
  startDate?: string;
  endDate?: string;
  venue?: string;
  location?: string;
  rawOptions?: unknown[];
  rawAddOns?: unknown[];
  selectedOption?: unknown;
  selectedOptions?: string[];
  isMultiOption?: boolean;
  onConfigureAddOns?: (payload: {
    lineItemId?: string;
    productId?: string;
    selectedAddOnIds: string[];
    quantity: number;
    selectedOption: string[];
  }) => void;
  lineItemId?: string;
  productId?: string;
  attachedAddOnIds?: string[];
  showAddOns?: boolean;
  showOptions?: boolean;
  isEarlyBird?: boolean;
}
 
// ─── Pencil / Edit icon ───────────────────────────────────────────────────────
 
const BasketItem = ({
  title,
  children,
  addOns = [],
  price,
  originalPrice,
  quantity = 1,
  onQuantityChange,
  onRemove,
  promoLabel: _promoLabel,
  promoValue: _promoValue,
  currency = CART_DEFAULTS.CURRENCY,
  locale = CART_DEFAULTS.LOCALE,
  startDate,
  endDate,
  venue,
  location,
  rawOptions: _rawOptions = [],
  rawAddOns = [],
  selectedOption = null,
  selectedOptions = [],
  isMultiOption: _isMultiOption = false,
  onConfigureAddOns,
  lineItemId,
  productId,
  attachedAddOnIds = [],
  showAddOns = true,
  showOptions = true,
  isEarlyBird = false,
}: BasketItemProps) => {
  const [isModalOpen, setIsModalOpen] = React.useState(false);
  const [showRemovePassModal, setShowRemovePassModal] = React.useState(false);
  const [modalSelectedAddOns, setModalSelectedAddOns] = React.useState<Record<string, boolean>>({});
  const [modalQuantity, setModalQuantity] = React.useState(quantity ?? 1);
  const [modalSelectedOptions, setModalSelectedOptions] = React.useState<string[]>(
    Array.isArray(selectedOptions) && selectedOptions.length > 0
      ? selectedOptions
      : typeof selectedOption === 'string'
        ? [selectedOption]
        : [],
  );
 
  const [initialModalState, setInitialModalState] = React.useState<{
    selectedAddOns: Record<string, boolean>;
    selectedOptions: string[];
    quantity: number;
  } | null>(null);
 
 
  // ── Derived flags ──────────────────────────────────────────────────────────
  const hasAddOns = addOns && addOns.length > 0;
  const hasAvailableAddOns = rawAddOns && rawAddOns.length > 0;
 
  // ── Quantity handlers ──────────────────────────────────────────────────────
  const handleDecrease = () => {
    if (quantity === 1) {
      setShowRemovePassModal(true);
    } else if (onQuantityChange) {
      onQuantityChange(quantity - 1);
    }
  };
 
  const handleIncrease = () => {
    if (onQuantityChange) onQuantityChange(quantity + 1);
  };
 
  const handleKeepPass = () => setShowRemovePassModal(false);
 
  const handleRemovePass = () => {
    setShowRemovePassModal(false);
    if (quantity > 1 && onQuantityChange) {
      onQuantityChange(quantity - 1);
    } else if (onRemove) {
      onRemove();
    }
  };
 
  // ── Modal handlers ─────────────────────────────────────────────────────────
  const handleOpenConfigureModal = () => {
    const initialSelections: Record<string, boolean> = {};
    (rawAddOns as Array<{ id: string }>)?.forEach((addon) => {
      initialSelections[addon.id] = attachedAddOnIds.includes(addon.id);
    });
    setModalSelectedAddOns(initialSelections);
    setModalQuantity(quantity);
 
    const initialOptions =
      Array.isArray(selectedOptions) && selectedOptions.length > 0
        ? selectedOptions
        : typeof selectedOption === 'string'
          ? [selectedOption]
          : [];
 
    setModalSelectedOptions(initialOptions);
    setInitialModalState({
      selectedAddOns: initialSelections,
      selectedOptions: initialOptions,
      quantity,
    });
    setIsModalOpen(true);
  };
 
  const handleCloseConfigureModal = () => {
    setIsModalOpen(false);
    setInitialModalState(null);
  };
 
  const hasChanges = () => {
    if (!initialModalState) return false;
    const addOnsChanged =
      JSON.stringify(modalSelectedAddOns) !== JSON.stringify(initialModalState.selectedAddOns);
    const optsChanged =
      JSON.stringify([...modalSelectedOptions].sort()) !==
      JSON.stringify([...(initialModalState.selectedOptions || [])].sort());
    const quantityChanged = modalQuantity !== initialModalState.quantity;
    return addOnsChanged || optsChanged || quantityChanged;
  };
 
  const handleAddOnSelectionChange = (addOnId: string, isSelected: boolean) => {
    setModalSelectedAddOns((prev) => ({ ...prev, [addOnId]: isSelected }));
  };
 
  const handleConfigureAddOns = (_selection: { addOns: unknown[]; options: unknown[] }) => {
    const selectedAddOnIds = Object.keys(modalSelectedAddOns).filter(
      (id) => modalSelectedAddOns[id],
    );
    if (onConfigureAddOns) {
      onConfigureAddOns({
        lineItemId,
        productId,
        selectedAddOnIds,
        quantity: modalQuantity,
        selectedOption: modalSelectedOptions,
      });
    }
    setIsModalOpen(false);
  };
 
  // ── Render ─────────────────────────────────────────────────────────────────
  return (
    <>
      {/* ── Outer card: 669.6×(152|248)px
            Default state : bg-white, border-t 1px #D0D5DD, shadow-sm (elevation-1), radius 6px
            Selected state: bg-[#F2F9FF], border-b 3px blue, shadow-sm (not yet implemented)
      ── */}
      <div
        className={[
          'bg-white shadow-[0_2px_8px_0_rgba(16,24,40,0.08),0_1px_3px_0_rgba(16,24,40,0.04)] rounded-[6px] overflow-hidden',
          'w-[335px] shrink-0 mx-auto tablet:p-[12px]',
          'tablet:w-[688px] tablet:mx-auto desktop:w-full desktop:mx-0',
          'desktop:max-w-[669.6px]',
          hasAddOns ? 'desktop:min-h-[248px]' : 'desktop:min-h-[152px]',
        ].join(' ')}
      >
        {/* ── First inner container: flexible for all screens ── */}
        <div className="relative tablet:min-h-[128px] px-[16px] py-[12px] tablet:py-[16px] tablet:px-[20px] flex flex-col justify-between gap-[10px] tablet:gap-4">
          {/* Top group: pass name row + date row */}
          <div className="flex flex-col min-h-[56px] gap-1 tablet:gap-1 w-[287px] tablet:w-full">
            {/* Row 1: Pass name + Configure Pass button */}
            <div className="flex items-start justify-between gap-2">
              {/* Pass name container */}
              <div className="flex-1 min-w-0 tablet:pl-0.5">
                <h3 className="m-0 break-words font-aleo text-[22px] font-semibold leading-7 tracking-[0] text-[#002B66]">
                  {(title || '').length > CART_COPY.MAX_PRODUCT_NAME_LENGTH
                    ? `${(title || '').substring(0, CART_COPY.MAX_PRODUCT_NAME_LENGTH)}...`
                    : title}
                </h3>
              </div>
 
              {/* Configure Pass button */}
              <button
                type="button"
                onClick={handleOpenConfigureModal}
                aria-label={CART_COPY.BASKET_ITEM_CONFIGURE_PASS_ARIA}
                className="shrink-0 inline-flex items-center justify-center rounded-[6px] border border-[#0A51C2] bg-white text-[#0A51C2] w-[24px] h-[24px] tablet:w-[127px] tablet:h-[26px] tablet:px-2 tablet:gap-1.5"
              >
                <span className="hidden tablet:inline text-xs font-bold leading-[18px] tracking-[0] whitespace-nowrap">
                  {CART_COPY.BASKET_ITEM_CONFIGURE_PASS}
                </span>
                <img
                  src={EditIcon}
                  className="w-[20px] h-[20px] tablet:w-4 tablet:h-4 shrink-0"
                  alt=""
                  aria-hidden="true"
                />
              </button>
            </div>
 
            {/* Row 2: Date · attendee type */}
            <div className="flex flex-col tablet:flex-row tablet:items-center tablet:flex-wrap gap-0 tablet:gap-x-[6px] tablet:gap-y-0.5 tablet:w-[462.6px]">
              {(startDate || endDate) && (
                <span className="flex items-center gap-1">
                  <img
                    src={CalendarIcon}
                    className="w-4 h-auto shrink-0"
                    alt=""
                    aria-hidden="true"
                  />
                  <DateRange
                    startDate={startDate}
                    endDate={endDate}
                    locale={locale}
                    format="short"
                    className="font-sans font-normal text-sm leading-6 tracking-[0px] text-[#002B66]/70"
                  />
                </span>
              )}
 
              {(venue || location) && (
                <>
                  <span className="hidden tablet:inline text-xs text-[#002B66]/40">·</span>
                  <span className="font-sans font-normal text-sm leading-6 tracking-[0px] text-[#002B66]/70">
                    {[venue, location].filter(Boolean).join(', ')}
                  </span>
                </>
              )}
 
              {children && (
                <span className="flex items-center gap-[4px]">
                  <img
                    src={UserAttributesIcon}
                    className="w-4 h-auto shrink-0"
                    alt=""
                    aria-hidden="true"
                  />
                  <span className="font-sans font-normal text-sm leading-6 tracking-[0px] text-[#002B66]/70">
                    {children}
                  </span>
                </span>
              )}
            </div>
          </div>
 
          {/* Row 3: Qty control + Add more pill + Price */}
          <div className="flex w-[287px] tablet:w-full min-h-[50px] items-end justify-between gap-2 mt-auto">
            {/* Left: qty control + pill */}
            <div className="flex items-center gap-2 min-w-0">
              {/* Shared QuantityControl MiniBasket */}
              <QuantityControl
                quantity={quantity}
                onDecrease={handleDecrease}
                onIncrease={handleIncrease}
                variant="MiniBasket"
              />
 
              {/* Add more passes pill (hidden on mobile) */}
              <div className="hidden tablet:flex shrink-0 items-center justify-center w-[274px] h-6 bg-[#F6F7F9] rounded-full overflow-hidden">
                <span className="w-[254px] h-[18px] text-center text-xs leading-[18px] text-[#002B66] font-normal truncate">
                  {/* TODO: qty and amount should come from event promotion config */}
                  {formatDiscountMessage(2, '$200')}
                </span>
              </div>
            </div>
 
            {/* Right: price stack */}
            <div className="flex flex-col items-end shrink-0 gap-0">
              {/* Struck-through original price */}
              {originalPrice && originalPrice !== price && (
                <span className="block line-through min-w-fit h-6 text-sm font-normal leading-6 text-[#002B66]/70 text-right">
                  {formatCurrency(originalPrice, currency, locale)}
                </span>
              )}
              {/* Main price */}
              <div className="flex items-center gap-3">
                {isEarlyBird && (
                  <img
                    src={PromoIcon}
                    className="w-[46px] h-[24px] shrink-0"
                    alt={CART_COPY.BASKET_ITEM_EARLY_BIRD_ALT}
                  />
                )}
                <span className="block min-w-fit h-[26px] text-lg font-semibold leading-[26px] text-[#002B66] text-right">
                  {formatCurrency(price, currency, locale)}
                </span>
              </div>
            </div>
          </div>
        </div>
 
        {/* ── Add-ons & extras section: visible only when addons selected ── */}
        {hasAddOns && (
          <div className="mt-[8px] mb-[8px] tablet:mt-3 tablet:mb-0 rounded-[6px] bg-[#F2F9FF] p-[16px] min-h-[84px] w-[319px] mx-auto tablet:w-auto tablet:mx-0">
            {/* Inner content container: 632px on tablet, gap 10px between label and items */}
            <div className="flex flex-col gap-[10px] tablet:w-[632px] desktop:w-auto">
              {/* Label: 12px/18px, Noto Sans 400, #002B66 */}
              <p className="font-sans font-normal text-xs leading-[18px] tracking-[0px] text-[#002B66] m-0">
                {CART_COPY.BASKET_ITEM_ADDONS_LABEL}
              </p>
              {/* Add-on items: each icon+name is a non-breaking unit, 12px between items, 2px icon-to-text */}
              <div className="flex flex-wrap gap-x-[12px] gap-y-[1px] tablet:gap-y-[4px]">
                {addOns.map((addOn, index) => (
                  <span key={addOn.id ?? index} className="inline-flex items-center gap-[2px]">
                    <img
                      src={AsteriskIcon}
                      className="w-5 h-5 shrink-0"
                      alt=""
                      aria-hidden="true"
                    />
                    <span className="font-sans font-medium text-sm leading-6 tracking-[0px] text-[#002B66]">
                      {addOn.name}
                    </span>
                  </span>
                ))}
              </div>
            </div>
          </div>
        )}
      </div>
 
      {/* Pass Configuration Modal */}
      {hasAvailableAddOns && (showAddOns || showOptions) && (
        <PassConfigurationModal
          isOpen={isModalOpen}
          onClose={handleCloseConfigureModal}
          title={title}
          description={startDate || endDate ? formatDateRange(startDate ?? null, endDate ?? null) : ''}
          price={Math.round(price * 100)}
          originalPrice={originalPrice != null ? Math.round(originalPrice * 100) : undefined}
          currency={currency}
          rawAddOns={rawAddOns as RawAddOnProduct[]}
          activeCurrency={currency}
          selectedAddOns={modalSelectedAddOns}
          onSelectionChange={handleAddOnSelectionChange}
          onAddToBasket={handleConfigureAddOns as (selection: { addOns: unknown[] }) => void}
          buttonText={CART_COPY.BASKET_ITEM_UPDATE_PASS}
          buttonIcon={ConfirmationIcon}
          quantity={modalQuantity}
          onQuantityChange={setModalQuantity}
          buttonDisabled={!hasChanges()}
          hidePassDetails={false}
          hideQuantity={true}
          showAddOns={showAddOns}
        />
      )}
 
      {/* Remove Pass Modal */}
      <RemovePassModal
        open={showRemovePassModal}
        onKeep={handleKeepPass}
        onRemove={handleRemovePass}
      />
    </>
  );
};
 
export default BasketItem;
