import React from 'react';
 
import { UI_SYMBOLS } from '../../../constants/checkout.constants';
import { PAYMENT_TEXTS } from '../../../constants/payment.constants';
import { useInvoicePayment } from '../../../hooks/payment-method/useInvoicePayment';
import type { InvoicePaymentFormProps } from '../../../types/payment-method/Payment.types';
import PaymentMethodItem from '../shared/PaymentMethodItem';
import PaymentTerms from '../shared/PaymentTerms';
 
 
import errorIcon from '@/assets/icons/FormErrorValidationIcon.svg';
import DisabledSecureIcon from '@/assets/icons/header/checkout/disabledSecure.svg';
import SecureIcon from '@/assets/icons/header/checkout/secure.svg';
import closeIcon from '@/assets/icons/invoice-confirmation/CloseIcon.svg';
import invoiceIcon from '@/assets/icons/invoice-confirmation/InvoiceIcon.svg';
import invoiceBgImage from '@/assets/icons/invoice-confirmation/InvoiceImage.svg';
import invoiceNoteIcon from '@/assets/icons/invoice-confirmation/InvoiceNoteIcon.svg';
import Button from '@/shared/components/Button';
import Loader from '@/shared/components/Loader';
 
 
const CONTAINER_WIDTHS = 'max-[767px]:w-[287px] max-[767px]:mx-auto min-[768px]:max-[1024px]:w-[608px] min-[1025px]:w-[590px]';
const CONTENT_WIDTHS = 'max-[767px]:w-[239px] min-[768px]:max-[1024px]:w-[560px] min-[1025px]:w-[541.6px]';
const INPUT_WIDTHS = 'max-[767px]:w-[207px] min-[768px]:max-[1024px]:w-[528px] min-[1025px]:w-[509.6px]';
 
const InvoicePaymentForm: React.FC<InvoicePaymentFormProps> = ({
    onPayLater,
    processing,
    error,
    expanded,
    onToggle,
    termsAccepted,
    setTermsAccepted,
}) => {
    const { email, additionalEmail, setAdditionalEmail, validationError, setValidationError, handlePayLater, isEmailValid } = useInvoicePayment({
        termsAccepted,
        onPayLater,
    });
    const [showAdditionalEmail, setShowAdditionalEmail] = React.useState(false);
 
    React.useEffect(() => {
        if (!expanded) {
            setShowAdditionalEmail(false);
            setAdditionalEmail('');
            setValidationError('');
        }
    }, [expanded, setAdditionalEmail, setValidationError]);
 
    return (
        <PaymentMethodItem
            expanded={expanded}
            onToggle={onToggle}
            className={`mb-6 font-nano ${!expanded ? '!p-0 !border-0 !w-full bg-transparent' : `${CONTAINER_WIDTHS} !border-[#1A6FE5]`}`}
        >
            {/* Header Content */}
            {!expanded ? (
                <div className={`flex items-center relative overflow-hidden border-[1px] border-[#DFE4EC] rounded-[6px] bg-white ${CONTAINER_WIDTHS} h-[98px]`}>
                    {/* Background Decoration Image */}
                    <img
                        src={invoiceBgImage}
                        alt=""
                        aria-hidden="true"
                        className="absolute w-[148.97px] h-[148.97px] pointer-events-none left-[65px] tablet:left-[386px] desktop:left-[368px] bottom-[-25.55px] z-20"
                    />
                    {/* Invoice Icon */}
                    <div className="ml-[24px] flex-shrink-0 relative z-10">
                        <img src={invoiceIcon} alt="Invoice" className="w-[24px] h-[24px]" />
                    </div>
                    {/* Texts */}
                    <div className="ml-[20px] flex flex-col justify-center gap-[2px] relative z-10">
                        <h3 className="text-[#002B66] text-[16px] leading-[24px] font-semibold whitespace-nowrap mb-0">
                            {PAYMENT_TEXTS.INVOICE.TITLE}
                        </h3>
                        <p className="text-[#002B66] opacity-70 text-[14px] leading-[24px] font-normal whitespace-nowrap mb-0">
                            {PAYMENT_TEXTS.INVOICE.SUBTITLE}
                        </p>
                    </div>
                </div>
            ) : (
                <div className="flex items-center relative overflow-hidden h-[98px] -mt-[20px] -mx-[20px] w-[calc(100%+40px)] rounded-t-[6px]">
                    {/* Background Decoration Image */}
                    <img
                        src={invoiceBgImage}
                        alt=""
                        aria-hidden="true"
                        className="absolute w-[148.97px] h-[148.97px] pointer-events-none left-[65px] tablet:left-[386px] desktop:left-[368px] bottom-[-25.55px] z-20"
                    />
                    {/* Invoice Icon */}
                    <div className="ml-[24px] flex-shrink-0 relative z-10">
                        <img src={invoiceIcon} alt="Invoice" className="w-[24px] h-[24px]" />
                    </div>
                    {/* Texts */}
                    <div className="ml-[20px] flex flex-col justify-center gap-[2px] relative z-10">
                        <h3 className="text-[#002B66] text-[16px] leading-[24px] font-semibold whitespace-nowrap mb-0">
                            {PAYMENT_TEXTS.INVOICE.TITLE}
                        </h3>
                        <p className="text-[#002B66] opacity-70 text-[14px] leading-[24px] font-normal whitespace-nowrap mb-0">
                            {PAYMENT_TEXTS.INVOICE.SUBTITLE}
                        </p>
                    </div>
                </div>
            )}
 
            {expanded && (
                <>
                    <hr className={`mb-0 mt-0 border-t-[1px] border-[#F0F2F5] mx-auto ${CONTENT_WIDTHS}`} />
 
                    <div className="mt-[24px] ml-[4px]">
                        <div className="flex flex-wrap items-center text-[#3B485F] text-[14px]">
                            <span className="w-[156px] h-[24px] font-normal leading-[24px] flex-shrink-0 whitespace-nowrap">{PAYMENT_TEXTS.INVOICE.EMAIL_WILL_BE_SENT}</span>
                            <span className="max-[767px]:ml-0 min-[768px]:ml-[4px] font-normal text-[#0F1624] text-[14px] leading-[24px] break-all">{email}</span>
                        </div>
 
                        {!showAdditionalEmail ? (
                            <button
                                onClick={() => setShowAdditionalEmail(true)}
                                aria-expanded={showAdditionalEmail}
                                className={`mt-[8px] text-[#1A6FE5] font-bold text-[12px] leading-[18px] h-[18px] text-left hover:underline focus:outline-2 focus:outline-offset-2 focus:outline-[#1A6FE5] active:opacity-80 transition-colors ${CONTENT_WIDTHS}`}
                            >
                                {PAYMENT_TEXTS.INVOICE.ADD_ADDITIONAL_EMAIL}
                            </button>
                        ) : (
                            <div className="mt-[24px]">
                                <label className="block text-[#3B485F] font-medium text-[14px] leading-[14px] w-[110px] h-[14px]">
                                    {PAYMENT_TEXTS.INVOICE.ADDITIONAL_EMAIL_LABEL}
                                </label>
                                <div className={`mt-[10px] flex items-center ${CONTENT_WIDTHS}`}>
                                    <input
                                        type="email"
                                        value={additionalEmail}
                                        onChange={(e) => {
                                            setAdditionalEmail(e.target.value);
                                            if (validationError) setValidationError('');
                                        }}
                                        placeholder={PAYMENT_TEXTS.INVOICE.EMAIL_PLACEHOLDER}
                                        className={`h-[40px] px-[12px] py-[8px] border rounded-[6px] focus:outline-none focus:border-[#030407] w-full ${INPUT_WIDTHS} ${validationError ? 'border-[#C63939]' : 'border-[#D0D5DD]'}`}
                                    />
                                    <button
                                        type="button"
                                        disabled={processing}
                                        aria-label="Remove additional email field"
                                        onClick={() => {
                                            setShowAdditionalEmail(false);
                                            setAdditionalEmail('');
                                            if (validationError) setValidationError('');
                                        }}
                                        className="ml-[12px] flex-shrink-0 disabled:opacity-50 disabled:cursor-not-allowed"
                                    >
                                        <img src={closeIcon} alt="" aria-hidden="true" className="w-[20px] h-[20px]" />
                                    </button>
                                </div>
                                {validationError && (
                                    <div className="flex items-center mt-[10px]">
                                        <img src={errorIcon} alt="Error" className="w-[16px] h-[16px]" />
                                        <span className="ml-[4px] text-[#C63939] font-normal text-[12px] h-[16px] leading-[16px]">{validationError}</span>
                                    </div>
                                )}
                            </div>
                        )}
                    </div>
 
                    {/* Terms & Conditions */}
                    <PaymentTerms
                        termsAccepted={termsAccepted}
                        setTermsAccepted={setTermsAccepted}
                        id={PAYMENT_TEXTS.SHARED.TERMS_CHECKBOX_INVOICE_ID}
                        className="mt-[24px] mb-[20px] ml-[4px]"
                    />
 
                    <div className={`bg-[#F6F7F9] rounded-[6px] pt-[14px] pl-[16px] pr-[16px] mb-[24px] ml-[4px] ${CONTENT_WIDTHS} max-[767px]:h-[238px] min-[768px]:max-[1024px]:h-[130px] min-[1025px]:h-[130px] flex gap-[6px]`}>
                        <img src={invoiceNoteIcon} alt="" aria-hidden="true" className="w-[20px] h-[20px] mt-[2px] flex-shrink-0" />
                        <div className="max-[767px]:w-[189px] min-[768px]:max-[1024px]:w-[510px] min-[1025px]:w-[492px] max-[767px]:h-[214px] min-[768px]:max-[1024px]:h-[106px] min-[1025px]:h-[106px]">
                            <p className="text-[12px] text-[#536279] mb-[12px] leading-[18px] font-normal m-0">
                                {PAYMENT_TEXTS.INVOICE.INSTRUCTIONS}
                            </p>
                            <p className="text-[12px] text-[#536279] leading-[18px] font-normal m-0">
                                {PAYMENT_TEXTS.INVOICE.WARNING}
                            </p>
                        </div>
                    </div>
 
                    <Button
                        variant="primary"
                        onClick={() => handlePayLater(showAdditionalEmail)}
                        disabled={processing || !email || !termsAccepted || !isEmailValid(email)}
                        className={`w-full ${CONTENT_WIDTHS} h-[48px] px-[20px] ml-[4px] rounded-[6px] flex items-center justify-center gap-[6px] font-noto !font-bold !text-[14px] leading-[20px] disabled:bg-[#EAECF0] disabled:!text-[#667085] disabled:opacity-100 disabled:cursor-not-allowed bg-[#030407] text-white hover:bg-black`}
                    >
                        {processing ? (
                            <Loader size="sm" variant="dark" inline={true} />
                        ) : (
                            <>
                                <span>{PAYMENT_TEXTS.INVOICE.PAY_LATER}</span>
                                <img
                                    src={(!isEmailValid(email) || processing) ? DisabledSecureIcon : SecureIcon}
                                    alt=""
                                    aria-hidden="true"
                                    className={`w-[20px] h-[20px] ${(!isEmailValid(email) || processing) ? UI_SYMBOLS.EMPTY : 'brightness-0 invert'}`}
                                />
                            </>
                        )}
                    </Button>
                    {error && (
                        <p className="text-red-500 text-sm mt-4 text-center">{error}</p>
                    )}
                </>
            )}
        </PaymentMethodItem>
    );
};
 
export default InvoicePaymentForm;
