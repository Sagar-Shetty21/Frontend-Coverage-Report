import React from 'react';
 
import { useEventContext } from '../../../../app/provider/useEventContext';
import { appConfig } from '../../../../config/app-config';
import { formatDateRange } from '../../../../shared/utils/dateUtil';
import { formatPrice } from '../../../../shared/utils/format';
import { groupItemsByDelegateRate } from '../../../cart/utils/cartGrouping';
import { DEFAULT_CURRENCY, DISPLAY_FALLBACK } from '../../constants/checkout.constants';
import { LINE_ITEMS } from '../../constants/orderSummary.constants';
import { LineItemsPreviewProps, TransformedLineItem, AddonItem } from '../../types/checkout.types';
 
import AttendeeTypeIcon from '@/assets/icons/checkout/AttendeeType.svg';
import CalendarIcon from '@/assets/icons/checkout/CalendarIcon.svg';
import TicketQuantityIcon from '@/assets/icons/checkout/TicketQuantity.svg';
 
const LineItemsPreview: React.FC<LineItemsPreviewProps> = ({ lineItems, showTaxBreakdown }) => {
    const locale = appConfig.defaults.locale;
    const { eventData } = useEventContext();
 
    const dateRange = formatDateRange(
        (eventData?.startDate as string | null) ?? null,
        (eventData?.endDate as string | null) ?? null
    );
 
    // Group items for display
    const groupedItems = React.useMemo(() => groupItemsByDelegateRate(lineItems as unknown as import('../../../cart/types/cartTypes').CTLineItem[]) as unknown as Array<{ key: string; items: TransformedLineItem[] }>, [lineItems]);
 
    const getMoneyData = (price: number | { centAmount?: number; currencyCode?: string } | { value?: { centAmount?: number; currencyCode?: string } } | undefined | null) => {
        if (typeof price === 'number') {
            return { centAmount: price, currencyCode: DEFAULT_CURRENCY };
        }
        if (!price) {
            return { centAmount: 0, currencyCode: DEFAULT_CURRENCY };
        }
        if ('value' in price && price.value) {
            return {
                centAmount: price.value.centAmount || 0,
                currencyCode: price.value.currencyCode || DEFAULT_CURRENCY
            };
        }
        const p = price as { centAmount?: number; currencyCode?: string };
        return {
            centAmount: p.centAmount || 0,
            currencyCode: p.currencyCode || DEFAULT_CURRENCY
        };
    };
 
    const renderTaxBreakdown = (item: TransformedLineItem | AddonItem, qty: number) => {
        if (!showTaxBreakdown) return null;
 
        const priceData = item.priceData;
 
        let subtotalVal = 0;
        let taxVal = 0;
        const price = (item as TransformedLineItem).price?.value || (item as AddonItem).price;
        const { currencyCode } = getMoneyData(price);
 
        if (priceData) {
            subtotalVal = (priceData.taxableBasis || (priceData.unitAmount || 0) - (priceData.unitTaxAmount || 0));
            taxVal = priceData.unitTaxAmount || 0;
        } else {
            return null;
        }
 
        const lineSubtotal = { centAmount: Math.round(subtotalVal * 100 * qty), currencyCode };
        const lineTax = { centAmount: Math.round(taxVal * 100 * qty), currencyCode };
 
        return (
            <div className="flex flex-col gap-[2px] text-[12px] font-normal text-white/70 italic mt-2">
                <div className="flex justify-between">
                    <span>{LINE_ITEMS.SUBTOTAL}</span>
                    <span>{formatPrice(lineSubtotal, locale)}</span>
                </div>
                <div className="flex justify-between">
                    <span>{LINE_ITEMS.TOTAL_TAX}</span>
                    <span>{formatPrice(lineTax, locale)}</span>
                </div>
            </div>
        );
    };
 
    return (
        <div className="flex flex-col">
            {groupedItems.map((group: { key: string; items: TransformedLineItem[] }) => (
                <React.Fragment key={group.key}>
                    {group.items.map((item: TransformedLineItem, idx: number) => {
                        const name = (item.name && (item.name[locale] || item.name['en'] || Object.values(item.name)[0])) || LINE_ITEMS.ITEM_FALLBACK;
 
                        const originalUnitPrice = item.price?.value || item.price;
                        const { centAmount: originalCentAmount, currencyCode } = getMoneyData(originalUnitPrice);
                        const qty = item.quantity || 1;
 
                        const totalPrice = {
                            centAmount: originalCentAmount,
                            currencyCode
                        };
 
                        const addons = item.addons || [];
 
                        return (
                            <div key={item.id || idx} className="w-full flex flex-col transition-all mb-[24px] last:mb-0">
                                {/* Metadata Row — icons 12×12, 4px gap within group, 8px between groups */}
                                <div className="flex items-center flex-wrap font-sans font-normal text-[12px] leading-[18px] text-[#FFFFFFCC] mb-[2px]">
                                    {/* Quantity */}
                                    <div className="flex items-center gap-[4px]">
                                        <img src={TicketQuantityIcon} alt="" className="w-[12px] h-[12px] shrink-0" />
                                        <span>{qty}</span>
                                    </div>
                                    {/* Date range — 8px gap from qty */}
                                    <div className="flex items-center gap-[4px] ml-[8px]">
                                        <img src={CalendarIcon} alt="" className="w-[12px] h-[12px] shrink-0" />
                                        <span className="whitespace-nowrap">{dateRange}</span>
                                    </div>
                                    {/* Attendee type — 8px gap from date */}
                                    {group.key !== LINE_ITEMS.OTHER_GROUP && (
                                        <div className="flex items-center gap-[4px] ml-[8px]">
                                            <img src={AttendeeTypeIcon} alt="" className="w-[12px] h-[12px] shrink-0" />
                                            <span>{group.key}</span>
                                        </div>
                                    )}
                                </div>
 
                                {/* Main Product Row */}
                                <div className="flex items-start justify-between gap-[12px] w-full min-h-[24px]">
                                    <span className="flex-1 min-w-0 font-sans font-semibold text-[18px] leading-[26px] text-white">
                                        {name}
                                    </span>
                                    <span className="w-[74px] shrink-0 font-sans font-semibold text-[16px] leading-[24px] text-white text-right">
                                        {totalPrice ? formatPrice(totalPrice, locale) : DISPLAY_FALLBACK}
                                    </span>
                                </div>
 
                                {renderTaxBreakdown(item, qty)}
 
                                {/* Addons */}
                                {addons.length > 0 && (
                                    <div className="flex flex-col mt-[8px] gap-[2px]">
                                        {[...new Map(addons.map((a: AddonItem) => [a.name, a])).values()].map((addon: AddonItem, addonIdx: number) => {
                                            const addonMoney = addon.price || addon.total;
                                            const { centAmount: addonCentAmount, currencyCode: addonCurrencyCode } = getMoneyData(addonMoney);
                                            const multiplied = {
                                                centAmount: addonCentAmount * qty,
                                                currencyCode: addonCurrencyCode
                                            };
                                            return (
                                                <div key={addonIdx} className="flex flex-col gap-[1px]">
                                                    <div className="flex items-start justify-between gap-[12px] w-full min-h-[24px]">
                                                        <span className="flex-1 min-w-0 font-sans font-normal text-[14px] leading-[24px] text-white/80">
                                                            {addon.name}
                                                        </span>
                                                        <span className="w-[70px] shrink-0 font-sans font-normal text-[14px] leading-[24px] text-white/80 text-right">
                                                            {formatPrice(multiplied, locale)}
                                                        </span>
                                                    </div>
                                                    {renderTaxBreakdown(addon, qty)}
                                                </div>
                                            );
                                        })}
                                    </div>
                                )}
                            </div>
                        );
                    })}
                </React.Fragment>
            ))}
        </div>
    );
};
 
export default LineItemsPreview;
