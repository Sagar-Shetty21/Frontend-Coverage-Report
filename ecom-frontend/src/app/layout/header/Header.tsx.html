import { FC, ReactElement } from 'react';
import { useLocation } from 'react-router-dom';
 
import EncryptedIcon from '../../../assets/icons/encryptedblue.svg';
import { useCart } from '../../../features/cart/hooks/useCart';
import ProgressBar from '../../../shared/components/ProgressBar';
 
import { HEADER_PAGE_TITLES, HEADER_SECURE_TEXTS } from './header.constants';
 
export interface HeaderProps {
    /** Override the automatically calculated page title */
    title?: string;
    /** Override the automatically calculated secure text label */
    secureLabel?: string;
    /** Override the automatically calculated progress step (0-4) */
    stepOverride?: number;
    /** Custom color for the title text (hex or tailwind class) */
    titleColor?: string;
    /** Force the header to be visible regardless of current path */
    forceVisible?: boolean;
}
 
const HeaderContent: FC<HeaderProps & { path: string }> = ({
    path,
    title,
    secureLabel,
    stepOverride,
    titleColor = 'text-[#002B66]'
}): ReactElement | null => {
    const isSuccessPage = /\/[^/]+\/(payment-success|invoice-confirmation|order)/.test(path);
    const isCartPage = path.endsWith('/cart');
    const isCheckoutPage = path.includes('/checkout');
    const isRegisterPage = path.includes('/register-pass');
 
    // Scenario logic for Titles and Secure text (if not overridden via props)
    let pageTitle = title;
    if (pageTitle === undefined) {
        if (isCartPage) pageTitle = HEADER_PAGE_TITLES.CART;
        else if (isCheckoutPage) pageTitle = HEADER_PAGE_TITLES.CHECKOUT;
        else if (isRegisterPage) pageTitle = HEADER_PAGE_TITLES.REGISTER;
        else if (isSuccessPage) pageTitle = HEADER_PAGE_TITLES.SUCCESS;
    }
 
    let displaySecureText = secureLabel;
    if (displaySecureText === undefined) {
        if (isCartPage || isCheckoutPage) displaySecureText = HEADER_SECURE_TEXTS.PAYMENT_PROTECTED;
        else if (isRegisterPage) displaySecureText = HEADER_SECURE_TEXTS.DETAILS_SECURE;
        else if (isSuccessPage) displaySecureText = HEADER_SECURE_TEXTS.SUCCESS;
    }
 
    // Determine current step for ProgressBar (if not overridden)
    let step: number | null = stepOverride !== undefined ? stepOverride : null;
    if (step === null) {
        if (isCartPage) step = 0;
        else if (isCheckoutPage) step = 2;
        else if (isRegisterPage) step = 3;
        else if (isSuccessPage) step = 4;
    }
 
 
 
    return (
        <div className="w-full flex flex-col font-noto h-[112px] tablet:h-[114px] bg-white border-b border-[#F0F2F5] px-[20px] tablet:px-[40px] desktop:px-[70px]">
            <div className="w-full h-full">
                {/* Title & Secure Row: Responsive Widths & Heights centered via mx-auto */}
                <div className="mx-auto flex items-center justify-between 
                    w-[335px] h-[62px] pt-[24px] pb-[12px]
                    tablet:w-[688px] tablet:h-[64px] 
                    desktop:w-[1140px] desktop:max-w-[1140px]">
                    <h1 className={`font-aleo text-[22px] font-semibold leading-[28px] ${titleColor}`}>
                        {pageTitle}
                    </h1>
 
                    {displaySecureText && (
                        <div className="flex items-center gap-[6px]">
                            <img src={EncryptedIcon} alt="" className="w-5 h-5 select-none" aria-hidden="true" />
                            <span className="font-noto text-[14px] font-semibold leading-[20px] text-[#002B66]">
                                {displaySecureText}
                            </span>
                        </div>
                    )}
                </div>
 
                {/* Progress Bar Container: Constrained to match title row widths */}
                {step !== null && (
                    <div className="mx-auto w-[335px] tablet:w-[688px] desktop:w-[1140px]">
                        <ProgressBar currentStep={step} />
                    </div>
                )}
            </div>
        </div>
    );
};
 
const CartPageHeader: FC<HeaderProps & { path: string }> = (props): ReactElement | null => {
    const { cart, loading } = useCart();
 
    // Hide header if cart is empty on the cart page (unless forced via props)
    if (props.title === undefined) {
        const isCartEmpty = !cart || !cart.lineItems || cart.lineItems.length === 0;
        if (loading || isCartEmpty) {
            return null;
        }
    }
 
    return <HeaderContent {...props} />;
};
 
const Header: FC<HeaderProps> = (props): ReactElement | null => {
    const location = useLocation();
    const path = location.pathname;
 
    // Visibility gating: Hide on SSO paths (e.g. ping-callback)
    const isVisible = props.forceVisible || /\/[^/]+\/(cart|checkout|register-pass|payment|payment-success|invoice-confirmation|order)(\/|$)/.test(path);
    if (!isVisible) return null;
 
    if (path.endsWith('/cart')) {
        return <CartPageHeader path={path} {...props} />;
    }
 
    return <HeaderContent path={path} {...props} />;
};
 
export default Header;
